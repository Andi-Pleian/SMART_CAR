82bool aggregateArrayCounts(uint8_t aArray[], uint8_t aMaxIndex, uint8_t *aShortIndex, uint8_t *aLongIndex) {
83 uint8_t tSum = 0;
84 uint16_t tWeightedSum = 0;
85 for (uint_fast8_t i = 0; i <= aMaxIndex; i++) {
86 uint8_t tCurrentDurations = aArray[i];
87 if (tCurrentDurations != 0) {
88 // Add it to sum and remove array content
89 tSum += tCurrentDurations;
90 tWeightedSum += (tCurrentDurations * i);
91 aArray[i] = 0;
92 }
93 if ((tCurrentDurations == 0 || i == aMaxIndex) && tSum != 0) {
94 // here we have a sum and a gap after the values
95 uint8_t tAggregateIndex = (tWeightedSum + (tSum / 2)) / tSum; // with rounding
96 aArray[tAggregateIndex] = tSum; // disabling this line increases code size by 2 - unbelievable!
97 // store aggregate for later decoding
98 if (*aShortIndex == 0) {
99 *aShortIndex = tAggregateIndex;
100 } else if (*aLongIndex == 0) {
101 *aLongIndex = tAggregateIndex;
102 } else {
103 // we have 3 bins => this is likely no pulse width or distance protocol. e.g. it can be RC5.
104 return false;
105 }
106 // initialize for next aggregation
107 tSum = 0;
108 tWeightedSum = 0;
109 }
110 }
111 return true;
112}

115 * Try to decode a pulse width or pulse distance protocol.
116 * 1. Analyze all space and mark length
117 * 2. Decide if we have an pulse width or distance protocol
118 * 3. Try to decode with the mark and space data found in step 1

122 uint8_t tDurationArray[DURATION_ARRAY_SIZE];
123
124 /*
125 * Accept only protocols with at least 8 bits
126 */
127 if (decodedIRData.rawDataPtr->rawlen < (2 * 8) + 4) {
128 IR_DEBUG_PRINT("PULSE_DISTANCE: ");
129 IR_DEBUG_PRINT("Data length=");
130 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
131 IR_DEBUG_PRINTLN(" is less than 20");
132 return false;
133 }
134
135 uint_fast8_t i;
136// Reset array
137 memset(tDurationArray, 0, sizeof(tDurationArray));
138
139 uint8_t tMaxDurationIndex = 0;
140 // Count space durations. Skip leading start and trailing stop bit.
141 for (i = 4; i < (uint_fast8_t) decodedIRData.rawDataPtr->rawlen - 2; i += 2) {
142 uint8_t tDurationTicks = decodedIRData.rawDataPtr->rawbuf[i];
143 if (tDurationTicks < sizeof(tDurationArray)) {
144 tDurationArray[tDurationTicks]++;
145 if (tMaxDurationIndex < tDurationTicks) {
146 tMaxDurationIndex = tDurationTicks;
147 }
148 }
149 }
150
151// aggregate counts to one duration bin
152 uint8_t tSpaceTicksShort = 0;
153 uint8_t tSpaceTicksLong = 0;
154 if (!aggregateArrayCounts(tDurationArray, tMaxDurationIndex, &tSpaceTicksShort, &tSpaceTicksLong)) {
155 IR_DEBUG_PRINT(F("PULSE_DISTANCE: "));
156 IR_DEBUG_PRINTLN(F("Space aggregation failed, more than 2 distinct duration values found"));
157 return false;
158 }
159
160#if defined(DEBUG)
161 Serial.println(F("Space:"));
162 printDurations(tDurationArray, tMaxDurationIndex);
163#endif
164
165 // Reset array
166 memset(tDurationArray, 0, sizeof(tDurationArray));
167
168 tMaxDurationIndex = 0;
169 // Count mark durations. Skip leading start and trailing stop bit.
170 for (i = 3; i < (uint_fast8_t) decodedIRData.rawDataPtr->rawlen - 2; i += 2) {
171 uint8_t tDurationTicks = decodedIRData.rawDataPtr->rawbuf[i];
172 if (tDurationTicks < sizeof(tDurationArray)) {
173 tDurationArray[tDurationTicks]++;
174 if (tMaxDurationIndex < tDurationTicks) {
175 tMaxDurationIndex = tDurationTicks;
176 }
177 }
178 }
179
180 uint8_t tMarkTicksShort = 0;
181 uint8_t tMarkTicksLong = 0;
182 if (!aggregateArrayCounts(tDurationArray, tMaxDurationIndex, &tMarkTicksShort, &tMarkTicksLong)) {

184 IR_DEBUG_PRINTLN(F("Mark aggregation failed, more than 2 distinct duration values found"));

201 if (tMarkTicksLong == 0) {
202 IR_DEBUG_PRINT(F("PULSE_DISTANCE: "));
203 IR_DEBUG_PRINTLN(F("Only 1 distinct duration value for each space and mark found"));
204 return false;
