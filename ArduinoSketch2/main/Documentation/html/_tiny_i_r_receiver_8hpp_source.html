1/*
2 * TinyIRReceiver.hpp
3 *
4 * Receives IR protocol data of NEC protocol using pin change interrupts.
5 * NEC is the protocol of most cheap remote controls for Arduino.
6 *
7 * No parity check is done!
8 * On a completely received IR command, the user function handleReceivedIRData(uint16_t aAddress, uint8_t aCommand, bool isRepetition)
9 * is called in Interrupt context but with interrupts being enabled to enable use of delay() etc.
10 * !!!!!!!!!!!!!!!!!!!!!!
11 * Functions called in interrupt context should be running as short as possible,
12 * so if you require longer action, save the data (address + command) and handle them in the main loop.

105#if defined(ESP8266) || defined(ESP32)
106void IRAM_ATTR IRPinChangeInterruptHandler(void)
107#else
108void IRPinChangeInterruptHandler(void)
109#endif
110{
111#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
112 digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
113#endif
114 /*
115 * Save IR input level
116 * Negative logic, true / HIGH means inactive / IR space, LOW / false means IR mark.
117 */
118 uint_fast8_t tIRLevel = digitalReadFast(IR_INPUT_PIN);
119
120#if !defined(NO_LED_FEEDBACK_CODE) && defined(IR_FEEDBACK_LED_PIN)
121 digitalWriteFast(IR_FEEDBACK_LED_PIN, !tIRLevel);
122#endif
123
124 /*
125 * 1. compute microseconds after last change
126 */
127 uint32_t tCurrentMicros = micros();
128 uint16_t tMicrosOfMarkOrSpace = tCurrentMicros - TinyIRReceiverControl.LastChangeMicros;
129 TinyIRReceiverControl.LastChangeMicros = tCurrentMicros;
130
131 uint8_t tState = TinyIRReceiverControl.IRReceiverState;
132
133#ifdef TRACE
134 Serial.print(tState);
135 Serial.print(' ');
136// Serial.print(F(" I="));
137// Serial.print(tIRLevel);
138// Serial.print(F(" D="));
139// Serial.print(tDeltaMicros);
140// Serial.println();
141#endif
142
143 if (tIRLevel == LOW)
144 {
145 /*
146 * We have a mark here
147 */
148 if (tMicrosOfMarkOrSpace > 2 * NEC_HEADER_MARK)
149 {
150 // timeout -> must reset state machine
151 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
152 }
153 if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_MARK)
154 {
155 // We are at the beginning of the header mark, check timing at the next transition
156 tState = IR_RECEIVER_STATE_WAITING_FOR_START_SPACE;
157 }
158
159 else if (tState == IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK)
160 {
161 if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_HEADER_SPACE)
162 && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_HEADER_SPACE))
163 {
164 /*
165 * We have a valid data header space here -> initialize data
166 */
167 TinyIRReceiverControl.IRRawDataBitCounter = 0;
168 TinyIRReceiverControl.IRRawData.ULong = 0;
169 TinyIRReceiverControl.IRRawDataMask = 1;
170 TinyIRReceiverControl.IRRepeatDetected = false;
171 tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
172 }
173 else if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_REPEAT_HEADER_SPACE)
174 && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_REPEAT_HEADER_SPACE)
175 && TinyIRReceiverControl.IRRawDataBitCounter >= NEC_BITS)
176 {
177 /*
178 * We have a repeat header here and no broken receive before -> set repeat flag
179 */
180 TinyIRReceiverControl.IRRepeatDetected = true;
181 tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
182 }
183 else
184 {
185 // This parts are optimized by the compiler into jumps to one code :-)
186 // Wrong length -> reset state
187 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
188 }
189 }
190
191 else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK)
192 {
193 // Check data space length
194 if (tMicrosOfMarkOrSpace >= lowerValue(NEC_ZERO_SPACE) && tMicrosOfMarkOrSpace <= upperValue(NEC_ONE_SPACE))
195 {
196 // We have a valid bit here
197 tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
198 if (tMicrosOfMarkOrSpace >= 2 * NEC_UNIT)
199 {
200 // we received a 1
201 TinyIRReceiverControl.IRRawData.ULong |= TinyIRReceiverControl.IRRawDataMask;
202 }
203 else
204 {
205 // we received a 0 - empty code for documentation
206 }
207 // prepare for next bit
208 TinyIRReceiverControl.IRRawDataMask = TinyIRReceiverControl.IRRawDataMask << 1;
209 TinyIRReceiverControl.IRRawDataBitCounter++;
210 }
211 else
212 {
213 // Wrong length -> reset state
214 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
215 }
216 }
217 else
218 {
219 // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
220 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
221 }
222 }
223
224 else
225 {
226 /*
227 * We have a space here
228 */
229 if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_SPACE)
230 {
231 /*
232 * Check length of header mark here
233 */
234 if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_HEADER_MARK)
235 && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_HEADER_MARK))
236 {
237 tState = IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK;
238 }
239 else
240 {
241 // Wrong length of header mark -> reset state
242 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
243 }
244 }
245
246 else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE)
247 {
248 // Check data mark length
249 if (tMicrosOfMarkOrSpace >= lowerValue(NEC_BIT_MARK) && tMicrosOfMarkOrSpace <= upperValue(NEC_BIT_MARK))
250 {
251 /*
252 * We have a valid mark here, check for transmission complete
253 */
254 if (TinyIRReceiverControl.IRRawDataBitCounter >= NEC_BITS || TinyIRReceiverControl.IRRepeatDetected)
255 {
256 /*
257 * Code complete -> call callback, no parity check!
258 */
259 // Reset state for new start
260 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
261#if !defined(ARDUINO_ARCH_MBED)
262 interrupts();
263#endif
264 /*
265 * Address reduction to 8 bit
266 */
267 if (TinyIRReceiverControl.IRRawData.UByte.LowByte
268 == (uint8_t) (~TinyIRReceiverControl.IRRawData.UByte.MidLowByte))
269 {
270 // standard 8 bit address NEC protocol
271 TinyIRReceiverControl.IRRawData.UByte.MidLowByte = 0; // Address is the first 8 bit
272 }
273
274 /*
275 * Call user provided callback here
276 */
277 handleReceivedTinyIRData(TinyIRReceiverControl.IRRawData.UWord.LowWord,
278 TinyIRReceiverControl.IRRawData.UByte.MidHighByte, TinyIRReceiverControl.IRRepeatDetected);
279
280 }
281 else
282 {
283 // not finished yet
284 tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK;
285 }
286 }
287 else
288 {
289 // Wrong length -> reset state
290 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
291 }
292 }
293 else
294 {
295 // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
296 tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
297 }
298 }
299
300 TinyIRReceiverControl.IRReceiverState = tState;
301#ifdef _IR_MEASURE_TIMING
302 digitalWriteFast(_IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
303#endif
304}

338#if defined(__AVR_ATtiny1616__) || defined(__AVR_ATtiny3216__) || defined(__AVR_ATtiny3217__)

341#elif !defined(__AVR__) || defined(TINY_RECEIVER_USE_ARDUINO_ATTACH_INTERRUPT)
342 // costs 112 bytes FLASH + 4bytes RAM

344# ifdef DEBUG

351# if defined(__AVR_ATtiny25__) || defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
352 // use PinChangeInterrupt no INT0 for pin PB2
353 PCMSK = _BV(IR_INPUT_PIN);
354 // clear interrupt bit
355 GIFR |= 1 << PCIF;
356 // enable interrupt on next change
357 GIMSK |= 1 << PCIE;
358
359# elif defined(__AVR_ATtiny87__) || defined(__AVR_ATtiny167__)
360# if defined(ARDUINO_AVR_DIGISPARKPRO)
361# if (IR_INPUT_PIN == 3)
362 // interrupt on any logical change
363 EICRA |= _BV(ISC00);
364 // clear interrupt bit
365 EIFR |= 1 << INTF0;
366 // enable interrupt on next change
367 EIMSK |= 1 << INT0;
368# elif (IR_INPUT_PIN == 9)
369 EICRA |= _BV(ISC10);
370 // clear interrupt bit
371 EIFR |= 1 << INTF1;
372 // enable interrupt on next change
373 EIMSK |= 1 << INT1;
374# else
375# error "IR_INPUT_PIN must be 9 or 3."
376# endif // if (IR_INPUT_PIN == 9)
377
378# else // defined(ARDUINO_AVR_DIGISPARKPRO)
379# if (IR_INPUT_PIN == 14)
380 // interrupt on any logical change
381 EICRA |= _BV(ISC00);
382 // clear interrupt bit
383 EIFR |= 1 << INTF0;
384 // enable interrupt on next change
385 EIMSK |= 1 << INT0;
386# elif (IR_INPUT_PIN == 3)
387 EICRA |= _BV(ISC10);
388 // clear interrupt bit
389 EIFR |= 1 << INTF1;
390 // enable interrupt on next change
391 EIMSK |= 1 << INT1;
392# else
393# error "IR_INPUT_PIN must be 14 or 3."
394# endif // if (IR_INPUT_PIN == 14)
395# endif
396
397# else // defined(__AVR_ATtiny25__)
398 /*
399 * ATmegas + ATtiny88 here
400 */
401# if (IR_INPUT_PIN == 2)
402 // interrupt on any logical change
403 EICRA |= _BV(ISC00);
404 // clear interrupt bit
405 EIFR |= 1 << INTF0;
406 // enable interrupt on next change
407 EIMSK |= 1 << INT0;
408# elif (IR_INPUT_PIN == 3)
409 EICRA |= _BV(ISC10);
410// clear interrupt bit
411 EIFR |= 1 << INTF1;
412// enable interrupt on next change
413 EIMSK |= 1 << INT1;
414# elif IR_INPUT_PIN == 4 || IR_INPUT_PIN == 5 || IR_INPUT_PIN == 6 || IR_INPUT_PIN == 7
415 //ATmega328 (Uno, Nano ) etc. Enable pin change interrupt 20 to 23 for port PD4 to PD7 (Arduino pin 4 to 7)
416 PCICR |= _BV(PCIE2);
417 PCMSK2 = digitalPinToBitMask(IR_INPUT_PIN);
418# elif IR_INPUT_PIN == 8 || IR_INPUT_PIN == 9 || IR_INPUT_PIN == 10 || IR_INPUT_PIN == 11 || IR_INPUT_PIN == 12 || IR_INPUT_PIN == 13
419 //ATmega328 (Uno, Nano ) etc. Enable pin change interrupt 0 to 5 for port PB0 to PB5 (Arduino pin 8 to 13)

421 PCMSK0 = digitalPinToBitMask(IR_INPUT_PIN);
422# elif IR_INPUT_PIN == A0 || IR_INPUT_PIN == A1 || IR_INPUT_PIN == A2 || IR_INPUT_PIN == A3 || IR_INPUT_PIN == A4 || IR_INPUT_PIN == A5
423 //ATmega328 (Uno, Nano ) etc. Enable pin change interrupt 8 to 13 for port PC0 to PC5 (Arduino pin A0 to A5)
424 PCICR |= _BV(PCIE1);
425 PCMSK1 = digitalPinToBitMask(IR_INPUT_PIN);
426# else
427# error "IR_INPUT_PIN not allowed."
428# endif // if (IR_INPUT_PIN == 2)
429# endif // defined(__AVR_ATtiny25__)
430#endif // ! defined(__AVR__) || defined(TINY_RECEIVER_USE_ARDUINO_ATTACH_INTERRUPT)
431}

514/*

523# if defined(ARDUINO_AVR_DIGISPARKPRO)
524ISR(INT0_vect) // Pin 3 / PB6 / INT0 is connected to USB+ on DigisparkPro boards
525# else
526ISR(INT1_vect) // Pin 3 global assignment
527# endif
528
529# elif (IR_INPUT_PIN == 9) && defined(ARDUINO_AVR_DIGISPARKPRO) // Digispark pro
530ISR(INT1_vect)
531
532# elif (IR_INPUT_PIN == 14) && (defined(__AVR_ATtiny87__) || defined(__AVR_ATtiny167__))// For AVR_ATtiny167 INT0 is on pin 14 / PB6
533ISR(INT0_vect)

535# elif (! defined(ISC10)) || ((defined(__AVR_ATtiny87__) || defined(__AVR_ATtiny167__)) && INT1_PIN != 3)
536// on ATtinyX5 we do not have a INT1_vect but we can use the PCINT0_vect
