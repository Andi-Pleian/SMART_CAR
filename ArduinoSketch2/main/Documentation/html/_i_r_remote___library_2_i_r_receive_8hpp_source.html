152void IRrecv::enableIRIn() {
153
154 noInterrupts();
155
156 // Setup pulse clock TickCounterForISR interrupt
157 timerConfigForReceive();
158 TIMER_ENABLE_RECEIVE_INTR; // Timer interrupt enable
159 TIMER_RESET_INTR_PENDING; // NOP for most platforms
160
161 // Initialize state machine state
162 resume();

164
165 // Set pin modes

200
201 if (irparams.OverflowFlag) {
202 // Copy overflow flag to decodedIRData.flags and reset it

401bool IRrecv::decodePulseWidthData(uint8_t aNumberOfBits, uint8_t aStartOffset, uint16_t aOneMarkMicros, uint16_t aZeroMarkMicros,
402 uint16_t aBitSpaceMicros, bool aMSBfirst) {
403
404 uint16_t *tRawBufPointer = &decodedIRData.rawDataPtr->rawbuf[aStartOffset];
405 uint32_t tDecodedData = 0;
406
407 if (aMSBfirst) {
408 for (uint_fast8_t i = 0; i < aNumberOfBits; i++) {
409 // Check for variable length mark indicating a 0 or 1
410 if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
411 tDecodedData = (tDecodedData << 1) | 1;
412 IR_TRACE_PRINT('1');
413 } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
414 tDecodedData = (tDecodedData << 1) | 0;
415 IR_TRACE_PRINT('0');
416 } else {
417 IR_DEBUG_PRINT(F("Mark="));
418 IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
419 IR_DEBUG_PRINT(F(" is not "));
420 IR_DEBUG_PRINT(aOneMarkMicros);
421 IR_DEBUG_PRINT(F(" or "));
422 IR_DEBUG_PRINT(aZeroMarkMicros);
423 IR_DEBUG_PRINT(' ');
424 return false;
425 }
426 tRawBufPointer++;
427
428 if (tRawBufPointer < &decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen]) {
429 // Assume that last space, which is not recorded, is correct, since we can not check it
430 // Check for constant length space
431 if (!matchSpace(*tRawBufPointer, aBitSpaceMicros)) {
432 IR_DEBUG_PRINT(F("Space="));
433 IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
434 IR_DEBUG_PRINT(F(" is not "));
435 IR_DEBUG_PRINT(aBitSpaceMicros);
436 IR_DEBUG_PRINT(' ');
437 return false;
438 }
439 tRawBufPointer++;
440 }
441 }
442 IR_TRACE_PRINTLN("");
443 } else {
444 for (uint32_t tMask = 1UL; aNumberOfBits > 0; tMask <<= 1, aNumberOfBits--) {
445
446 // Check for variable length mark indicating a 0 or 1
447 if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
448 tDecodedData |= tMask; // set the bit
449 IR_TRACE_PRINT('1');
450 } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
451 // do not set the bit
452 IR_TRACE_PRINT('0');
453 } else {
454 IR_DEBUG_PRINT(F("Mark="));
455 IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
456 IR_DEBUG_PRINT(F(" is not "));
457 IR_DEBUG_PRINT(aOneMarkMicros);
458 IR_DEBUG_PRINT(F(" or "));
459 IR_DEBUG_PRINT(aZeroMarkMicros);
460 IR_DEBUG_PRINT(' ');
461 return false;
462 }
463 tRawBufPointer++;
464
465 if (tRawBufPointer < &decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen]) {
466 // Assume that last space, which is not recorded, is correct, since we can not check it
467 // Check for constant length space
468 if (!matchSpace(*tRawBufPointer, aBitSpaceMicros)) {
469 IR_DEBUG_PRINT(F("Space="));
470 IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
471 IR_DEBUG_PRINT(F(" is not "));
472 IR_DEBUG_PRINT(aBitSpaceMicros);
473 IR_DEBUG_PRINT(' ');
474 return false;
475 }
476 tRawBufPointer++;
477 }
478 }
479 IR_TRACE_PRINTLN("");
480 }
481 decodedIRData.decodedRawData = tDecodedData;
482 return true;

605uint8_t IRrecv::getBiphaselevel() {
606 uint8_t tLevelOfCurrentInterval; // 0 (SPACE) or 1 (MARK)
607
608 if (sBiphaseDecodeRawbuffOffset >= decodedIRData.rawDataPtr->rawlen) {
609 return SPACE; // After end of recorded buffer, assume space.
610 }
611
612 tLevelOfCurrentInterval = (sBiphaseDecodeRawbuffOffset) & 1; // on odd rawbuf offsets we have mark timings
613
614 /*
615 * Setup data if sUsedTimingIntervals is 0
616 */
617 if (sUsedTimingIntervals == 0) {
618 uint16_t tCurrentTimingWith = decodedIRData.rawDataPtr->rawbuf[sBiphaseDecodeRawbuffOffset];
619 uint16_t tMarkExcessCorrection = (tLevelOfCurrentInterval == MARK) ? MARK_EXCESS_MICROS : -MARK_EXCESS_MICROS;
620
621 if (matchTicks(tCurrentTimingWith, (sBiphaseTimeUnit) + tMarkExcessCorrection)) {
622 sCurrentTimingIntervals = 1;
623 } else if (matchTicks(tCurrentTimingWith, (2 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
624 sCurrentTimingIntervals = 2;
625 } else if (matchTicks(tCurrentTimingWith, (3 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
626 sCurrentTimingIntervals = 3;
627 } else {
628 return -1;
629 }
630 }
631
632 // We use another interval from tCurrentTimingIntervals
633 sUsedTimingIntervals++;
634
635 // keep track of current timing offset
636 if (sUsedTimingIntervals >= sCurrentTimingIntervals) {
637 // we have used all intervals of current timing, switch to next timing value
638 sUsedTimingIntervals = 0;
639 sBiphaseDecodeRawbuffOffset++;
640 }
641
642 IR_TRACE_PRINTLN(tLevelOfCurrentInterval);
643
644 return tLevelOfCurrentInterval;
645}

687bool IRrecv::decodeHash() {
688 unsigned long hash = FNV_BASIS_32; // the result is the same no matter if we use a long or unsigned long variable
689
690// Require at least 6 samples to prevent triggering on noise
691 if (decodedIRData.rawDataPtr->rawlen < 6) {
692 return false;
693 }
694#if RAW_BUFFER_LENGTH <= 254 // saves around 75 bytes program space and speeds up ISR
695 uint8_t i;
696#else
697 uint16_t i;
698#endif
699 for (i = 1; (i + 2) < decodedIRData.rawDataPtr->rawlen; i++) {
700 uint8_t value = compare(decodedIRData.rawDataPtr->rawbuf[i], decodedIRData.rawDataPtr->rawbuf[i + 2]);
701 // Add value into the hash
702 hash = (hash * FNV_PRIME_32) ^ value;
703 }
704
705 decodedIRData.decodedRawData = hash;
706 decodedIRData.numberOfBits = 32;
707 decodedIRData.protocol = UNKNOWN;
708
709 return true;
710}

1342{
1343#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
1344 digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1345#endif
1346// 7 - 8.5 us for ISR body (without pushes and pops) for ATmega328 @16MHz

1353#else
1354 uint8_t tIRInputLevel = (uint8_t) digitalRead(irparams.IRReceivePin);
1355#endif
1356
1357 /*
1358 * Increase TickCounter and clip it at maximum 0xFFFF / 3.2 seconds at 50 us ticks
1359 */
1360 if (irparams.TickCounterForISR < 0xFFFF) {
1361 irparams.TickCounterForISR++; // One more 50uS tick
1362 }
1363
1364 /*
1365 * Due to a ESP32 compiler bug https://github.com/espressif/esp-idf/issues/1552 no switch statements are possible for ESP32
1366 * So we change the code to if / else if
1367 */
1368// switch (irparams.StateForISR) {
1369//......................................................................
1370 if (irparams.StateForISR == IR_REC_STATE_IDLE) { // In the middle of a gap or just resumed (and maybe in the middle of a transmission
1371 if (tIRInputLevel == INPUT_MARK) {
1372 // check if we did not start in the middle of a transmission by checking the minimum length of leading space
1373 if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
1374 // Gap just ended; Record gap duration + start recording transmission
1375 // Initialize all state machine variables
1376#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
1377// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1378#endif

1382 irparams.StateForISR = IR_REC_STATE_MARK;
1383 } // otherwise stay in idle state
1384 irparams.TickCounterForISR = 0;// reset counter in both cases
1385 }
1386
1387 } else if (irparams.StateForISR == IR_REC_STATE_MARK) { // Timing mark
1388 if (tIRInputLevel != INPUT_MARK) { // Mark ended; Record time
1389#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
1390// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1391#endif

1393 irparams.StateForISR = IR_REC_STATE_SPACE;
1394 irparams.TickCounterForISR = 0;
1395 }
1396
1397 } else if (irparams.StateForISR == IR_REC_STATE_SPACE) { // Timing space
1398 if (tIRInputLevel == INPUT_MARK) { // Space just ended; Record time
1399 if (irparams.rawlen >= RAW_BUFFER_LENGTH) {
1400 // Flag up a read OverflowFlag; Stop the state machine
1401 irparams.OverflowFlag = true;
1402 irparams.StateForISR = IR_REC_STATE_STOP;

1404#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
1405// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1406#endif

1408 irparams.StateForISR = IR_REC_STATE_MARK;
1409 }
1410 irparams.TickCounterForISR = 0;
1411
1412 } else if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
1413 /*
1414 * Current code is ready for processing!
1415 * We received a long space, which indicates gap between codes.
1416 * Switch to IR_REC_STATE_STOP
1417 * Don't reset TickCounterForISR; keep counting width of next leading space
1418 */
1419 irparams.StateForISR = IR_REC_STATE_STOP;
1420 }
1421 } else if (irparams.StateForISR == IR_REC_STATE_STOP) {
1422 /*
1423 * Complete command received
1424 * stay here until resume() is called, which switches state to IR_REC_STATE_IDLE
1425 */
1426#if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
1427// digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1428#endif
1429 if (tIRInputLevel == INPUT_MARK) {
1430 // Reset gap TickCounterForISR, to prepare for detection if we are in the middle of a transmission after call of resume()
1431 irparams.TickCounterForISR = 0;
1432 }
1433 }
1434
1435#if !defined(NO_LED_FEEDBACK_CODE)
1436 if (FeedbackLEDControl.LedFeedbackEnabled) {
1437 setFeedbackLED(tIRInputLevel == INPUT_MARK);
1438 }
1439#endif
1440
1441#ifdef _IR_MEASURE_TIMING
1442 digitalWriteFast(_IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
1443#endif
1444}
