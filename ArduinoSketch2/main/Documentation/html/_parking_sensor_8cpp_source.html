124void ParkingSensor::Task() {
125 if ((this->getDistanceToObject() <= PS_1ST_STAGE_CM) && (this->getDistanceToObject() > PS_2ST_STAGE_CM)) {
126
127 this->buzzer->Buzz(BUZZER_1ST_STAGE_DELAY_MS);
128 } else if ((this->getDistanceToObject() <= PS_2ST_STAGE_CM) && (this->getDistanceToObject() > PS_3ST_STAGE_CM)) {
129
130 this->buzzer->Buzz(BUZZER_2ST_STAGE_DELAY_MS);
131 } else if ((this->getDistanceToObject() <= PS_3ST_STAGE_CM) && (this->getDistanceToObject() > PS_4ST_STAGE_CM)) {
132
133 this->buzzer->Buzz(BUZZER_3ST_STAGE_DELAY_MS);
134 } else if ((this->getDistanceToObject() <= PS_4ST_STAGE_CM) && (this->getDistanceToObject() > PS_5ST_STAGE_CM)) {
135
136 this->buzzer->Buzz(BUZZER_4ST_STAGE_DELAY_MS);
137 } else if ((this->getDistanceToObject() <= PS_5ST_STAGE_CM) && (this->getDistanceToObject() > PS_MIN_DISTANCE_CM)) {
138
139 this->buzzer->Buzz(BUZZER_5ST_STAGE_DELAY_MS);
140 } else if (this->getDistanceToObject() <= PS_MIN_DISTANCE_CM) {
141
142 this->buzzer->Buzz(BUZZER_MIN_STAGE_DELAY_MS);
143 }
144}
