41//==============================================================================
42// SSSS OOO N N Y Y
43// S O O NN N Y Y
44// SSS O O N N N Y
45// S O O N NN Y
46// SSSS OOO N N Y
47//==============================================================================
48// see https://www.sbprojects.net/knowledge/ir/sirc.php
49// Here http://picprojects.org.uk/projects/sirc/ it is claimed, that many Sony remotes repeat each frame a minimum of 3 times
50// LSB first, start bit + 7 command + 5 to 13 address, no stop bit
51//
52#define SONY_ADDRESS_BITS 5
53#define SONY_COMMAND_BITS 7
54#define SONY_EXTRA_BITS 8
55#define SONY_BITS_MIN (SONY_COMMAND_BITS + SONY_ADDRESS_BITS) // 12 bits
56#define SONY_BITS_15 (SONY_COMMAND_BITS + SONY_ADDRESS_BITS + 3) // 15 bits
57#define SONY_BITS_MAX (SONY_COMMAND_BITS + SONY_ADDRESS_BITS + SONY_EXTRA_BITS) // 20 bits == SIRCS_20_PROTOCOL
58#define SONY_UNIT 600 // 24 periods of 40kHz
59
60#define SONY_HEADER_MARK (4 * SONY_UNIT) //2400
61#define SONY_ONE_MARK (2 * SONY_UNIT) // 1200
62#define SONY_ZERO_MARK SONY_UNIT
63#define SONY_SPACE SONY_UNIT
64
65#define SONY_AVERAGE_DURATION 21000 // SONY_HEADER_MARK + SONY_SPACE + 12 * 2,5 * SONY_UNIT // 2.5 because we assume more zeros than ones
66#define SONY_REPEAT_PERIOD 45000 // Commands are repeated every 45 ms (measured from start to start) for as long as the key on the remote control is held down.
67#define SONY_REPEAT_SPACE (SONY_REPEAT_PERIOD - SONY_AVERAGE_DURATION) // 24 ms
68
69/*
70 * Repeat commands should be sent in a 45 ms raster.
71 * There is NO delay after the last sent command / repeat!
72 * @param numberOfBits if == 20 send 13 address bits otherwise only 5 address bits
73 */
74void IRsend::sendSony(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, uint8_t numberOfBits) {
75 // Set IR carrier frequency
76 enableIROut(SONY_KHZ); // 40 kHz
77
78 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
79 while (tNumberOfCommands > 0) {
80
81 // Header
82 mark(SONY_HEADER_MARK);
83 space(SONY_SPACE);
84
85 // send 7 command bits LSB first
86 sendPulseDistanceWidthData(SONY_ONE_MARK, SONY_SPACE, SONY_ZERO_MARK, SONY_SPACE, aCommand, SONY_COMMAND_BITS,
87 PROTOCOL_IS_LSB_FIRST);
88 // send 5, 8, 13 address bits LSB first
89 sendPulseDistanceWidthData(SONY_ONE_MARK, SONY_SPACE, SONY_ZERO_MARK, SONY_SPACE, aAddress,
90 (numberOfBits - SONY_COMMAND_BITS), PROTOCOL_IS_LSB_FIRST);
91
92 tNumberOfCommands--;
93 // skip last delay!
94 if (tNumberOfCommands > 0) {
95 // send repeated command in a 45 ms raster
96 delay(SONY_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
97 }
98 }
99}

103bool IRrecv::decodeSony() {
104
105 // Check header "mark"
106 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], SONY_HEADER_MARK)) {
107 return false;
108 }
109
110 // Check we have enough data. +2 for initial gap and start bit mark and space minus the last/MSB space. NO stop bit! 26, 32, 42
111 if (decodedIRData.rawDataPtr->rawlen != (2 * SONY_BITS_MIN) + 2 && decodedIRData.rawDataPtr->rawlen != (2 * SONY_BITS_MAX) + 2
112 && decodedIRData.rawDataPtr->rawlen != (2 * SONY_BITS_15) + 2) {
113 // ??? IR_TRACE_PRINT since I saw this too often
114 IR_DEBUG_PRINT("Sony: ");
115 IR_DEBUG_PRINT("Data length=");
116 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
117 IR_DEBUG_PRINTLN(" is not 12, 15 or 20");
118 return false;
119 }
120 // Check header "space"
121 if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], SONY_SPACE)) {
122 IR_DEBUG_PRINT("Sony: ");
123 IR_DEBUG_PRINTLN("Header space length is wrong");
124 return false;
125 }
126
127 if (!decodePulseWidthData((decodedIRData.rawDataPtr->rawlen - 1) / 2, 3, SONY_ONE_MARK, SONY_ZERO_MARK, SONY_SPACE, PROTOCOL_IS_LSB_FIRST)) {
128 IR_DEBUG_PRINT("Sony: ");
129 IR_DEBUG_PRINTLN("Decode failed");
130 return false;
131 }
132
133 // Success
134// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
135 uint8_t tCommand = decodedIRData.decodedRawData & 0x7F; // first 7 bits
136 uint16_t tAddress = decodedIRData.decodedRawData >> 7; // next 5 or 8 or 13 bits
137
138 /*
139 * Check for repeat
140 */
141 if (decodedIRData.rawDataPtr->rawbuf[0] < (SONY_REPEAT_PERIOD / MICROS_PER_TICK)) {
142 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
143 }
144 decodedIRData.command = tCommand;
145 decodedIRData.address = tAddress;
146 decodedIRData.numberOfBits = (decodedIRData.rawDataPtr->rawlen - 1) / 2;
147 decodedIRData.protocol = SONY;
148
149 return true;
150}

152#define SONY_DOUBLE_SPACE_USECS 500 // usually see 713 - not using ticks as get number wrap around
153bool IRrecv::decodeSonyMSB(decode_results *aResults) {
154 long data = 0;
155 uint8_t bits = 0;
156 unsigned int offset = 0; // Dont skip first space, check its size
157
158 if (aResults->rawlen < (2 * SONY_BITS_MIN) + 2) {
159 return false;
160 }
161
162 // Some Sony's deliver repeats fast after first
163 // unfortunately can't spot difference from of repeat from two fast clicks
164 if (aResults->rawbuf[0] < (SONY_DOUBLE_SPACE_USECS / MICROS_PER_TICK)) {
165 IR_DEBUG_PRINTLN("IR Gap found");
166 aResults->bits = 0;
167 aResults->value = 0xFFFFFFFF;
168 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
169 decodedIRData.protocol = SONY;
170 return true;
171 }
172 offset++;
173
174 // Check header "mark"
175 if (!matchMark(aResults->rawbuf[offset], SONY_HEADER_MARK)) {
176 return false;
177 }
178 offset++;
179
180 // MSB first - Not compatible to standard, which says LSB first :-(
181 while (offset + 1 < aResults->rawlen) {
182
183 // First check for the constant space length, we do not have a space at the end of raw data
184 // we are lucky, since the start space is equal the data space.
185 if (!matchSpace(aResults->rawbuf[offset], SONY_SPACE)) {
186 return false;
187 }
188 offset++;
189
190 // bit value is determined by length of the mark
191 if (matchMark(aResults->rawbuf[offset], SONY_ONE_MARK)) {
192 data = (data << 1) | 1;
193 } else if (matchMark(aResults->rawbuf[offset], SONY_ZERO_MARK)) {
194 data = (data << 1) | 0;
195 } else {
196 return false;
197 }
198 offset++;
199 bits++;
200
201 }
202
203 aResults->bits = bits;
204 aResults->value = data;
205 aResults->decode_type = SONY;
206 decodedIRData.protocol = SONY;
207 return true;
208}
