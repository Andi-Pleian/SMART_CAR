46//==============================================================================
47// RRRR CCCC 55555
48// R R C 5
49// RRRR C 5555
50// R R C 5
51// R R CCCC 5555
52//==============================================================================
53//
54// see: https://www.sbprojects.net/knowledge/ir/rc5.php
55// https://en.wikipedia.org/wiki/Manchester_code
56// mark->space => 0
57// space->mark => 1
58// MSB first 1 start bit, 1 field bit, 1 toggle bit + 5 bit address + 6 bit command (6 bit command plus one field bit for RC5X), no stop bit
59// duty factor is 25%,
60//
61#define RC5_ADDRESS_BITS 5
62#define RC5_COMMAND_BITS 6
63#define RC5_COMMAND_FIELD_BIT 1
64#define RC5_TOGGLE_BIT 1
65
66#define RC5_BITS (RC5_COMMAND_FIELD_BIT + RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS) // 13
67
68#define RC5_UNIT 889 // 32 periods of 36 kHz (888.8888)
69
70#define MIN_RC5_MARKS ((RC5_BITS + 1) / 2) // 7
71
72#define RC5_DURATION (15L * RC5_UNIT) // 13335
73#define RC5_REPEAT_PERIOD (128L * RC5_UNIT) // 113792
74#define RC5_REPEAT_SPACE (RC5_REPEAT_PERIOD - RC5_DURATION) // 100 ms
75
79void IRsend::sendRC5(uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aEnableAutomaticToggle) {
80 // Set IR carrier frequency
81 enableIROut(RC5_RC6_KHZ);
82
83 uint16_t tIRData = ((aAddress & 0x1F) << RC5_COMMAND_BITS);
84
85 if (aCommand < 0x40) {
86 // set field bit to lower field / set inverted upper command bit
87 tIRData |= 1 << (RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS);
88 } else {
89 // let field bit zero
90 aCommand &= 0x3F;
91 }
92
93 tIRData |= aCommand;
94
95 if (aEnableAutomaticToggle) {
96 if (sLastSendToggleValue == 0) {
97 sLastSendToggleValue = 1;
98 // set toggled bit
99 tIRData |= 1 << (RC5_ADDRESS_BITS + RC5_COMMAND_BITS);
100 } else {
101 sLastSendToggleValue = 0;
102 }
103 }
104
105 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
106 while (tNumberOfCommands > 0) {
107
108 // start bit is sent by sendBiphaseData
109 sendBiphaseData(RC5_UNIT, tIRData, RC5_BITS);
110
111 tNumberOfCommands--;
112 // skip last delay!
113 if (tNumberOfCommands > 0) {
114 // send repeated command in a fixed raster
115 delay(RC5_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
116 }
117 }
118}

132bool IRrecv::decodeRC5() {
133 uint8_t tBitIndex;
134 uint32_t tDecodedRawData = 0;
135
136 // Set Biphase decoding start values
137 initBiphaselevel(1, RC5_UNIT); // Skip gap space
138
139 // Check we have the right amount of data (11 to 26). The +2 is for initial gap and start bit mark.
140 if (decodedIRData.rawDataPtr->rawlen < MIN_RC5_MARKS + 2 && decodedIRData.rawDataPtr->rawlen > ((2 * RC5_BITS) + 2)) {
141 // no debug output, since this check is mainly to determine the received protocol
142 IR_DEBUG_PRINT(F("RC5: "));
143 IR_DEBUG_PRINT("Data length=");
144 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
145 IR_DEBUG_PRINTLN(" is not between 11 and 26");
146 return false;
147 }
148
149// Check start bit, the first space is included in the gap
150 if (getBiphaselevel() != MARK) {
151 IR_DEBUG_PRINT(F("RC5: "));
152 IR_DEBUG_PRINTLN("first getBiphaselevel() is not MARK");
153 return false;
154 }
155
156 /*
157 * Get data bits - MSB first
158 */
159 for (tBitIndex = 0; sBiphaseDecodeRawbuffOffset < decodedIRData.rawDataPtr->rawlen; tBitIndex++) {
160 uint8_t tStartLevel = getBiphaselevel();
161 uint8_t tEndLevel = getBiphaselevel();
162
163 if ((tStartLevel == SPACE) && (tEndLevel == MARK)) {
164 // we have a space to mark transition here
165 tDecodedRawData = (tDecodedRawData << 1) | 1;
166 } else if ((tStartLevel == MARK) && (tEndLevel == SPACE)) {
167 // we have a mark to space transition here
168 tDecodedRawData = (tDecodedRawData << 1) | 0;
169 } else {
170 // IR_TRACE_PRINT since I saw this too often
171 IR_DEBUG_PRINT(F("RC5: "));
172 IR_DEBUG_PRINTLN(F("Decode failed"));
173 return false;
174 }
175 }
176
177 // Success
178 decodedIRData.numberOfBits = tBitIndex; // must be RC5_BITS
179
180 LongUnion tValue;
181 tValue.ULong = tDecodedRawData;
182 decodedIRData.decodedRawData = tDecodedRawData;
183 decodedIRData.command = tValue.UByte.LowByte & 0x3F;
184 decodedIRData.address = (tValue.UWord.LowWord >> RC5_COMMAND_BITS) & 0x1F;
185
186 // Get the inverted 7. command bit for RC5X, the inverted value is always 1 for RC5 and serves as a second start bit.
187 if ((tValue.UWord.LowWord & (1 << (RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS))) == 0) {
188 decodedIRData.command += 0x40;
189 }
190
191 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
192 if (tValue.UByte.MidLowByte & 0x8) {
193 decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST;
194 }
195
196 // check for repeat
197 if (decodedIRData.rawDataPtr->rawbuf[0] < (RC5_REPEAT_PERIOD / MICROS_PER_TICK)) {
198 decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
199 }
200
201 decodedIRData.protocol = RC5;
202 return true;
203}

222
223#define RC6_LEADING_BIT 1
224#define RC6_MODE_BITS 3 // never seen others than all 0 for Philips TV
225#define RC6_TOGGLE_BIT 1 // toggles at every key press. Can be used to distinguish repeats from 2 key presses.
226#define RC6_ADDRESS_BITS 8
227#define RC6_COMMAND_BITS 8

231#define RC6_UNIT 444 // 16 periods of 36 kHz (444.4444)

233#define RC6_HEADER_MARK (6 * RC6_UNIT) // 2666
234#define RC6_HEADER_SPACE (2 * RC6_UNIT) // 889

236#define RC6_TRAILING_SPACE (6 * RC6_UNIT) // 2666
237#define MIN_RC6_MARKS 4 + ((RC6_ADDRESS_BITS + RC6_COMMAND_BITS) / 2) // 12, 4 are for preamble

244void IRsend::sendRC6(uint32_t aRawData, uint8_t aNumberOfBitsToSend) {
245// Set IR carrier frequency
246 enableIROut(RC5_RC6_KHZ);
247
248// Header
249 mark(RC6_HEADER_MARK);
250 space(RC6_HEADER_SPACE);
251
252// Start bit
253 mark(RC6_UNIT);
254 space(RC6_UNIT);
255
256// Data MSB first
257 uint32_t mask = 1UL << (aNumberOfBitsToSend - 1);
258 for (uint_fast8_t i = 1; mask; i++, mask >>= 1) {
259 // The fourth bit we send is the "double width toggle bit"
260 unsigned int t = (i == 4) ? (RC6_UNIT * 2) : (RC6_UNIT);
261 if (aRawData & mask) {
262 mark(t);
263 space(t);
264 } else {
265 space(t);
266 mark(t);
267 }
268 }
269}

349bool IRrecv::decodeRC6() {
350 uint8_t tBitIndex;
351 uint32_t tDecodedRawData = 0;
352
353 // Check we have the right amount of data (). The +3 for initial gap, start bit mark and space
354 if (decodedIRData.rawDataPtr->rawlen < MIN_RC6_MARKS + 3 && decodedIRData.rawDataPtr->rawlen > ((2 * RC6_BITS) + 3)) {
355 IR_DEBUG_PRINT(F("RC6: "));
356 IR_DEBUG_PRINT("Data length=");
357 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
358 IR_DEBUG_PRINTLN(" is not between 15 and 45");
359 return false;
360 }
361
362 // Check header "mark" and "space", this must be done for repeat and data
363 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], RC6_HEADER_MARK)
364 || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], RC6_HEADER_SPACE)) {
365 // no debug output, since this check is mainly to determine the received protocol
366 IR_DEBUG_PRINT(F("RC6: "));
367 IR_DEBUG_PRINTLN("Header mark or space length is wrong");
368 return false;
369 }
370
371 // Set Biphase decoding start values
372 initBiphaselevel(3, RC6_UNIT); // Skip gap-space and start-bit mark + space
373
374// Process first bit, which is known to be a 1 (mark->space)
375 if (getBiphaselevel() != MARK) {
376 IR_DEBUG_PRINT(F("RC6: "));
377 IR_DEBUG_PRINTLN("first getBiphaselevel() is not MARK");
378 return false;
379 }
380 if (getBiphaselevel() != SPACE) {
381 IR_DEBUG_PRINT(F("RC6: "));
382 IR_DEBUG_PRINTLN("second getBiphaselevel() is not SPACE");
383 return false;
384 }
385
386 for (tBitIndex = 0; sBiphaseDecodeRawbuffOffset < decodedIRData.rawDataPtr->rawlen; tBitIndex++) {
387 uint8_t tStartLevel; // start level of coded bit
388 uint8_t tEndLevel; // end level of coded bit
389
390 tStartLevel = getBiphaselevel();
391 if (tBitIndex == 3) {
392 // Toggle bit is double wide; make sure second half is equal first half
393 if (tStartLevel != getBiphaselevel()) {
394 IR_DEBUG_PRINT(F("RC6: "));
395 IR_DEBUG_PRINTLN(F("Toggle mark or space length is wrong"));
396 return false;
397 }
398 }
399
400 tEndLevel = getBiphaselevel();
401 if (tBitIndex == 3) {
402 // Toggle bit is double wide; make sure second half matches
403 if (tEndLevel != getBiphaselevel()) {
404 IR_DEBUG_PRINT(F("RC6: "));
405 IR_DEBUG_PRINTLN(F("Toggle mark or space length is wrong"));
406 return false;
407 }
408 }
409
410 /*
411 * Determine tDecodedRawData bit value by checking the transition type
412 */
413 if ((tStartLevel == MARK) && (tEndLevel == SPACE)) {
414 // we have a mark to space transition here
415 tDecodedRawData = (tDecodedRawData << 1) | 1; // inverted compared to RC5
416 } else if ((tStartLevel == SPACE) && (tEndLevel == MARK)) {
417 // we have a space to mark transition here
418 tDecodedRawData = (tDecodedRawData << 1) | 0;
419 } else {
420 IR_DEBUG_PRINT(F("RC6: "));
421 IR_DEBUG_PRINTLN(F("Decode failed"));
422 // we have no transition here or one level is -1 -> error
423 return false; // Error
424 }
425 }
426
427// Success
428 decodedIRData.numberOfBits = tBitIndex;
429
430 LongUnion tValue;
431 tValue.ULong = tDecodedRawData;
432 decodedIRData.decodedRawData = tDecodedRawData;
433
434 if (tBitIndex < 36) {
435 // RC6
436 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
437 decodedIRData.command = tValue.UByte.LowByte;
438 decodedIRData.address = tValue.UByte.MidLowByte;
439 // Check for toggle flag
440 if ((tValue.UByte.MidHighByte & 1) != 0) {
441 decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST;
442 }
443 } else {
444 // RC6A
445 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST | IRDATA_FLAGS_EXTRA_INFO;
446 if ((tValue.UByte.MidLowByte & 0x80) != 0) {
447 decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST | IRDATA_FLAGS_EXTRA_INFO;
448 }
449 tValue.UByte.MidLowByte &= 0x87F; // mask toggle bit
450 decodedIRData.command = tValue.UByte.LowByte;
451 decodedIRData.address = tValue.UByte.MidLowByte;
452 // get extra info
453 decodedIRData.extra = tValue.UWord.HighWord;
454 }
455
456 // check for repeat, do not check toggle bit yet
457 if (decodedIRData.rawDataPtr->rawbuf[0] < ((RC6_REPEAT_SPACE + (RC6_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
458 decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
459 }
460
461 decodedIRData.protocol = RC6;
462 return true;
463}

468void IRsend::sendRC5(uint32_t data, uint8_t nbits) {
469 // Set IR carrier frequency
470 enableIROut(RC5_RC6_KHZ);
471
472 // Start
473 mark(RC5_UNIT);
474 space(RC5_UNIT);
475 mark(RC5_UNIT);
476
477 // Data - Biphase code MSB first
478 for (uint32_t mask = 1UL << (nbits - 1); mask; mask >>= 1) {
479 if (data & mask) {
480 space(RC5_UNIT); // 1 is space, then mark
481 mark(RC5_UNIT);
482 } else {
483 mark(RC5_UNIT);
484 space(RC5_UNIT);
485 }
486 }
487}

492void IRsend::sendRC5ext(uint8_t addr, uint8_t cmd, bool toggle) {
493// Set IR carrier frequency
494 enableIROut(RC5_RC6_KHZ);
495
496 uint8_t addressBits = 5;
497 uint8_t commandBits = 7;
498// unsigned long nbits = addressBits + commandBits;
499
500// Start
501 mark(RC5_UNIT);
502
503// Bit #6 of the command part, but inverted!
504 uint8_t cmdBit6 = (1UL << (commandBits - 1)) & cmd;
505 if (cmdBit6) {
506 // Inverted (1 -> 0 = mark-to-space)
507 mark(RC5_UNIT);
508 space(RC5_UNIT);
509 } else {
510 space(RC5_UNIT);
511 mark(RC5_UNIT);
512 }
513 commandBits--;
514
515// Toggle bit
516 static int toggleBit = 1;
517 if (toggle) {
518 if (toggleBit == 0) {
519 toggleBit = 1;
520 } else {
521 toggleBit = 0;
522 }
523 }
524 if (toggleBit) {
525 space(RC5_UNIT);
526 mark(RC5_UNIT);
527 } else {
528 mark(RC5_UNIT);
529 space(RC5_UNIT);
530 }
531
532// Address
533 for (uint_fast8_t mask = 1UL << (addressBits - 1); mask; mask >>= 1) {
534 if (addr & mask) {
535 space(RC5_UNIT); // 1 is space, then mark
536 mark(RC5_UNIT);
537 } else {
538 mark(RC5_UNIT);
539 space(RC5_UNIT);
540 }
541 }
542
543// Command
544 for (uint_fast8_t mask = 1UL << (commandBits - 1); mask; mask >>= 1) {
545 if (cmd & mask) {
546 space(RC5_UNIT); // 1 is space, then mark
547 mark(RC5_UNIT);
548 } else {
549 mark(RC5_UNIT);
550 space(RC5_UNIT);
551 }
552 }
