42// LSB first, 1 start bit + 16 bit address + 8 bit data + 8 bit inverted data + 1 stop bit.
43// see: https://www.sbprojects.net/knowledge/ir/nec.php
44
45#define NEC_ADDRESS_BITS 16 // 16 bit address or 8 bit address and 8 bit inverted address
46#define NEC_COMMAND_BITS 16 // Command and inverted command
47
48#define NEC_BITS (NEC_ADDRESS_BITS + NEC_COMMAND_BITS)
49#define NEC_UNIT 560
50
51#define NEC_HEADER_MARK (16 * NEC_UNIT) // 9000
52#define NEC_HEADER_SPACE (8 * NEC_UNIT) // 4500
53
54#define NEC_BIT_MARK NEC_UNIT
55#define NEC_ONE_SPACE (3 * NEC_UNIT) // 1690
56#define NEC_ZERO_SPACE NEC_UNIT
57
58#define NEC_REPEAT_HEADER_SPACE (4 * NEC_UNIT) // 2250
59#define NEC_REPEAT_PERIOD 110000 // Not used yet - Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
