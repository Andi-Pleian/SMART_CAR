255void IRsend::sendRaw(const uint16_t aBufferWithMicroseconds[], uint_fast8_t aLengthOfBuffer, uint_fast8_t aIRFrequencyKilohertz) {
256// Set IR carrier frequency
257 enableIROut(aIRFrequencyKilohertz);
258
259 /*
260 * Raw data starts with a mark.
261 */
262 for (uint_fast8_t i = 0; i < aLengthOfBuffer; i++) {
263 if (i & 1) {
264 // Odd
265 space(aBufferWithMicroseconds[i]);
266 } else {
267 mark(aBufferWithMicroseconds[i]);
268 }
269 }

347void IRsend::sendPulseDistanceWidthData(unsigned int aOneMarkMicros, unsigned int aOneSpaceMicros, unsigned int aZeroMarkMicros,
348 unsigned int aZeroSpaceMicros, uint32_t aData, uint8_t aNumberOfBits, bool aMSBfirst, bool aSendStopBit) {
349
350 if (aMSBfirst) { // Send the MSB first.
351 // send data from MSB to LSB until mask bit is shifted out
352 for (uint32_t tMask = 1UL << (aNumberOfBits - 1); tMask; tMask >>= 1) {
353 if (aData & tMask) {
354 IR_TRACE_PRINT('1');
355 mark(aOneMarkMicros);
356 space(aOneSpaceMicros);
357 } else {
358 IR_TRACE_PRINT('0');
359 mark(aZeroMarkMicros);
360 space(aZeroSpaceMicros);
361 }
362 }
363 } else { // Send the Least Significant Bit (LSB) first / MSB last.
364 for (uint_fast8_t bit = 0; bit < aNumberOfBits; bit++, aData >>= 1)
365 if (aData & 1) { // Send a 1
366 IR_TRACE_PRINT('1');
367 mark(aOneMarkMicros);
368 space(aOneSpaceMicros);
369 } else { // Send a 0
370 IR_TRACE_PRINT('0');
371 mark(aZeroMarkMicros);
372 space(aZeroSpaceMicros);
373 }
374 }
375 if (aSendStopBit) {
376 IR_TRACE_PRINT('S');
377 mark(aZeroMarkMicros); // seems like this is used for stop bits
378 }
379 IR_TRACE_PRINTLN("");
380}
381
382/*
383 * Sends Biphase data MSB first
384 * Always send start bit, do not send the trailing space of the start bit
385 * 0 -> mark+space
386 * 1 -> space+mark
387 * The output always ends with a space
388 */
389void IRsend::sendBiphaseData(unsigned int aBiphaseTimeUnit, uint32_t aData, uint_fast8_t aNumberOfBits) {
390
391// do not send the trailing space of the start bit
392 mark(aBiphaseTimeUnit);
393
394 IR_TRACE_PRINT('S');
395 uint8_t tLastBitValue = 1; // Start bit is a 1
396
397// Data - Biphase code MSB first
398 for (uint32_t tMask = 1UL << (aNumberOfBits - 1); tMask; tMask >>= 1) {
399 if (aData & tMask) {
400 IR_TRACE_PRINT('1');
401 space(aBiphaseTimeUnit);
402 mark(aBiphaseTimeUnit);
403 tLastBitValue = 1;
404
405 } else {
406 IR_TRACE_PRINT('0');
407#if defined(SEND_PWM_BY_TIMER) || defined(USE_NO_SEND_PWM)
408 if (tLastBitValue) {
409 // Extend the current mark in order to generate a continuous signal without short breaks
410 delayMicroseconds(aBiphaseTimeUnit);
411 } else {
412 mark(aBiphaseTimeUnit);
413 }
414#else
415 (void) tLastBitValue; // to avoid compiler warnings
416 mark(aBiphaseTimeUnit); // can not eventually delay here, we must call mark to generate the signal
417#endif
418 space(aBiphaseTimeUnit);
419 tLastBitValue = 0;
420 }
421 }
422 IR_TRACE_PRINTLN("");
423}

431void IRsend::mark(unsigned int aMarkMicros) {
432
433#if defined(SEND_PWM_BY_TIMER)
434# if !defined(NO_LED_FEEDBACK_CODE)
435 setFeedbackLED(true);
436# endif
437 ENABLE_SEND_PWM_BY_TIMER; // Enable timer or ledcWrite() generated PWM output
438 customDelayMicroseconds(aMarkMicros);
439 IRLedOff();
440# if !defined(NO_LED_FEEDBACK_CODE)
441 setFeedbackLED(false);
442# endif
443
444#elif defined(USE_NO_SEND_PWM)
445# if !defined(NO_LED_FEEDBACK_CODE)
446 setFeedbackLED(true);
447# endif
448# if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && !defined(OUTPUT_OPEN_DRAIN)
449 pinModeFast(sendPin, OUTPUT); // active state for mimicking open drain
450# else
451 digitalWriteFast(sendPin, LOW); // Set output to active low.
452# endif
453
454 customDelayMicroseconds(aMarkMicros);
455 IRLedOff();
456# if !defined(NO_LED_FEEDBACK_CODE)
457 setFeedbackLED(false);
458# endif
459
460#else
461 unsigned long startMicros = micros();
462 unsigned long nextPeriodEnding = startMicros;
463 unsigned long tMicros;
464# if !defined(NO_LED_FEEDBACK_CODE)
465 bool FeedbackLedIsActive = false;
466# endif
467
468 do {
469// digitalToggleFast(_IR_TIMING_TEST_PIN);
470 // Output the PWM pulse
471 noInterrupts(); // do not let interrupts extend the short on period
472# if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN)
473# if defined(OUTPUT_OPEN_DRAIN)
474 digitalWriteFast(sendPin, LOW); // active state for open drain
475# else
476 pinModeFast(sendPin, OUTPUT); // active state for mimicking open drain
477# endif
478# else
479# if defined(IR_SEND_PIN)
480 digitalWriteFast(IR_SEND_PIN, HIGH); // 3.5 us from FeedbackLed on to pin setting. 5.7 us from call of mark() to pin setting incl. setting of feedback pin.
481# else
482 digitalWrite(sendPin, HIGH); // 4.3 us from do{ to pin setting
483# endif
484# endif
485 delayMicroseconds(periodOnTimeMicros); // this is normally implemented by a blocking wait
486
487 // Output the PWM pause
488# if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && !defined(OUTPUT_OPEN_DRAIN)
489# if defined(OUTPUT_OPEN_DRAIN)
490 digitalWriteFast(sendPin, HIGH); // Set output to inactive high.
491# else
492 pinModeFast(sendPin, INPUT); // inactive state to mimic open drain
493# endif
494
495# else
496 digitalWriteFast(sendPin, LOW);
497# endif
498 interrupts(); // Enable interrupts -to keep micros correct- for the longer off period 3.4 us until receive ISR is active (for 7 us + pop's)
499
500# if !defined(NO_LED_FEEDBACK_CODE)
501 /*
502 * Delayed call of setFeedbackLED() to get better timing
503 */
504 if (!FeedbackLedIsActive) {
505 FeedbackLedIsActive = true;
506 setFeedbackLED(true);
507 }
508# endif
509 /*
510 * Pause timing
511 */
512 nextPeriodEnding += periodTimeMicros;
513 do {
514 tMicros = micros(); // we have only 4 us resolution for AVR @16MHz
515 // check for aMarkMicros to be gone
516 unsigned int tDeltaMicros = tMicros - startMicros;
517#if defined(__AVR__)
518// tDeltaMicros += (160 / CLOCKS_PER_MICRO); // adding this once increases program size !
519# if !defined(NO_LED_FEEDBACK_CODE)
520 if (tDeltaMicros >= aMarkMicros - (30 + (112 / CLOCKS_PER_MICRO))) { // 30 to be constant. Using periodTimeMicros increases program size too much.
521 // reset feedback led in the last pause before end
522 setFeedbackLED(false);
523 }
524# endif
525 if (tDeltaMicros >= aMarkMicros - (112 / CLOCKS_PER_MICRO)) { // To compensate for call duration - 112 is an empirical value
526#else
527 if (tDeltaMicros >= aMarkMicros) {
528#endif
529 return;
530 }
531// digitalToggleFast(_IR_TIMING_TEST_PIN); // 3.0 us per call @16MHz
532 } while (tMicros < nextPeriodEnding); // 3.4 us @16MHz
533 } while (true);
534# endif
535}

542void IRsend::IRLedOff() {
543#if defined(SEND_PWM_BY_TIMER)
544 DISABLE_SEND_PWM_BY_TIMER; // Disable PWM output
545#elif defined(USE_NO_SEND_PWM)
546# if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && !defined(OUTPUT_OPEN_DRAIN)
547 digitalWriteFast(sendPin, LOW); // prepare for all next active states.
548 pinModeFast(sendPin, INPUT); // inactive state for open drain
549# else
550 digitalWriteFast(sendPin, HIGH); // Set output to inactive high.

552#else
553# if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN)
554# if defined(OUTPUT_OPEN_DRAIN)
555 digitalWriteFast(sendPin, HIGH); // Set output to inactive high.
556# else
557 pinModeFast(sendPin, INPUT); // inactive state to mimic open drain
558# endif
559# else
560 digitalWriteFast(sendPin, LOW);
561# endif
562#endif
563#if !defined(NO_LED_FEEDBACK_CODE)
564 setFeedbackLED(false);
565#endif
566}

603void IRsend::enableIROut(uint8_t aFrequencyKHz) {
604#if defined(SEND_PWM_BY_TIMER)
605 TIMER_DISABLE_RECEIVE_INTR; // Just in case receiving was active
606 timerConfigForSend(aFrequencyKHz);
607
608#elif defined(USE_NO_SEND_PWM)
609 (void) aFrequencyKHz;
610
611#else
612 periodTimeMicros = (1000U + (aFrequencyKHz / 2)) / aFrequencyKHz; // rounded value -> 26 for 38.46 kHz, 27 for 37.04 kHz, 25 for 40 kHz.
613# if defined(IR_SEND_PIN)
614 periodOnTimeMicros = (((periodTimeMicros * IR_SEND_DUTY_CYCLE) + 50) / 100U); // +50 for rounding -> 830/100 for 30% and 16 MHz
615# else
616 periodOnTimeMicros = (((periodTimeMicros * IR_SEND_DUTY_CYCLE) + 50 - (PULSE_CORRECTION_NANOS / 10)) / 100U); // +50 for rounding -> 530/100 for 30% and 16 MHz
617# endif
618#endif
619
620#if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN)
621# if defined(OUTPUT_OPEN_DRAIN)
622 pinMode(sendPin, OUTPUT_OPEN_DRAIN); // the only place where this mode is set for sendPin
623# endif // the mode INPUT for mimicking open drain is set at IRLedOff()
624
625#else
626 pinModeFast(sendPin, OUTPUT);
627#endif
628 IRLedOff(); // When not sending, we want it low/inactive
629}
