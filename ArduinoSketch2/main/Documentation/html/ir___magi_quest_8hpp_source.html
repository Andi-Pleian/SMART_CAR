48#if !defined (DOXYGEN)
49// MagiQuest packet is both Wand ID and magnitude of swish and flick
50union magiquest_t {
51 uint64_t llword;
52 struct {
53 uint16_t magnitude;
54 uint32_t wand_id;
55 uint8_t padding;
56 uint8_t scrap; // just to pad the struct out to 64 bits so we can union with llword
57 } cmd;
58};
59#endif // !defined (DOXYGEN)

61#define MAGIQUEST_MAGNITUDE_BITS (sizeof(uint16_t) * 8) // magiquest_t.cmd.magnitude
62#define MAGIQUEST_WAND_ID_BITS (sizeof(uint32_t) * 8) // magiquest_t.cmd.wand_id
63#define MAGIQUEST_PADDING_BITS (sizeof(uint8_t) * 8) // magiquest_t.cmd.padding
64
65#define MAGIQUEST_PERIOD 1150 // Length of time a full MQ "bit" consumes (1100 - 1200 usec)
66#define MAGIQUEST_BITS (MAGIQUEST_MAGNITUDE_BITS + MAGIQUEST_WAND_ID_BITS) // Size of the command itself
67
68// The total size of a packet is the sum of all 3 expected fields * 2 to support start/stop bits
69#define MAGIQUEST_PACKET_SIZE (2 * (MAGIQUEST_BITS + MAGIQUEST_PADDING_BITS))
70
71/*
72 * 0 = 25% mark & 75% space across 1 period
73 * 1150 * 0.25 = 288 usec mark
74 * 1150 - 288 = 862 usec space
75 * 1 = 50% mark & 50% space across 1 period
76 * 1150 * 0.5 = 575 usec mark
77 * 1150 - 575 = 575 usec space
78 */
79#define MAGIQUEST_UNIT (MAGIQUEST_PERIOD / 4)
80
81#define MAGIQUEST_ONE_MARK (2 * MAGIQUEST_UNIT) // 576
82#define MAGIQUEST_ONE_SPACE (2 * MAGIQUEST_UNIT) // 576
83#define MAGIQUEST_ZERO_MARK MAGIQUEST_UNIT
84#define MAGIQUEST_ZERO_SPACE (3 * MAGIQUEST_UNIT) // 864

102 MAGIQUEST_ONE_MARK, MAGIQUEST_ONE_SPACE, MAGIQUEST_ZERO_MARK, MAGIQUEST_ZERO_SPACE, magnitude, MAGIQUEST_MAGNITUDE_BITS,
103 PROTOCOL_IS_MSB_FIRST,
104 SEND_STOP_BIT);
105}

109/*
110 * decodes a 56 bit result, which is not really compatible with standard decoder layout
111 */
112bool IRrecv::decodeMagiQuest() {
113 magiquest_t data; // Somewhere to build our code
114 unsigned int offset = 1; // Skip the gap reading
115
116 unsigned int mark_;
117 unsigned int space_;
118 unsigned int ratio_;
119
120#ifdef DEBUG
121 char bitstring[(MAGIQUEST_PACKET_SIZE + 1)];
122 bitstring[MAGIQUEST_PACKET_SIZE] = '\0';
123#endif
124
125 // Check we have the right amount of data
126 if (decodedIRData.rawDataPtr->rawlen != MAGIQUEST_PACKET_SIZE) {
127 IR_DEBUG_PRINT("MagiQuest: Bad packet length - got ");
128 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
129 IR_DEBUG_PRINT(", expected ");
130 IR_DEBUG_PRINTLN(MAGIQUEST_PACKET_SIZE);
131 return false;
132 }
133
134 // Read the bits in
135 data.llword = 0;
136 while (offset < (MAGIQUEST_PACKET_SIZE - 1)) {
137 mark_ = decodedIRData.rawDataPtr->rawbuf[offset++];
138 space_ = decodedIRData.rawDataPtr->rawbuf[offset++];
139 ratio_ = space_ / mark_;
140
141 IR_TRACE_PRINT("MagiQuest: mark=");
142 IR_TRACE_PRINT(mark_ * MICROS_PER_TICK);
143 IR_TRACE_PRINT(" space=");
144 IR_TRACE_PRINT(space_ * MICROS_PER_TICK);
145 IR_TRACE_PRINT(" ratio=");
146 IR_TRACE_PRINTLN(ratio_);
147
148 if (matchMark(space_ + mark_, MAGIQUEST_PERIOD)) {
149 if (ratio_ > 1) {
150 // It's a 0
151 data.llword <<= 1;
152#ifdef DEBUG
153 bitstring[(offset / 2) - 1] = '0';
154#endif
155 } else {
156 // It's a 1
157 data.llword = (data.llword << 1) | 1;
158#ifdef DEBUG
159 bitstring[(offset / 2) - 1] = '1';
160#endif
161 }
162 } else {
163 IR_DEBUG_PRINTLN("MATCH_MARK failed");
164 return false;
165 }
166 }
167 IR_DEBUG_PRINTLN(bitstring);
168
169 // Success
170 decodedIRData.protocol = MAGIQUEST;
171 decodedIRData.numberOfBits = offset / 2;
172 decodedIRData.flags = IRDATA_FLAGS_EXTRA_INFO;
173 decodedIRData.extra = data.cmd.magnitude;
174 decodedIRData.decodedRawData = data.cmd.wand_id;
175
176 return true;
177}
178#endif
