43//==============================================================================
44// L EEEEEE EEEE OOOO
45// L E E O O
46// L EEEE E EEE O O
47// L E E E O O
48// LLLLLL EEEEEE EEEE OOOO
49//==============================================================================
50// from LEGO Power Functions RC Manual 26.02.2010 Version 1.20
51// https://github.com/jurriaan/Arduino-PowerFunctions/raw/master/LEGO_Power_Functions_RC_v120.pdf
52// https://oberguru.net/elektronik/ir/codes/lego_power_functions_train.lircd.conf
53//
54// To ensure correct detection of IR messages six 38 kHz cycles are transmitted as mark.
55// Low bit consists of 6 cycles of IR and 10 “cycles” of pause,
56// high bit of 6 cycles IR and 21 “cycles” of pause and start bit of 6 cycles IR and 39 “cycles” of pause.
57// Low bit range 316 - 526 us
58// High bit range 526 – 947 us
59// Start/stop bit range 947 – 1579 us
60
61// If tm is the maximum message length (16ms) and Ch is the channel number, then
62// The delay before transmitting the first message is: (4 – Ch)*tm
63// The time from start to start for the next 2 messages is: 5*tm
64// The time from start to start for the following messages is: (6 + 2*Ch)*tm
65
66// Supported Devices
67// LEGO Power Functions IR Receiver 8884
68
69// MSB first, 1 start bit + 4 bit channel, 4 bit mode + 4 bit command + 4 bit parity + 1 stop bit.
70#define LEGO_CHANNEL_BITS 4
71#define LEGO_MODE_BITS 4
72#define LEGO_COMMAND_BITS 4
73#define LEGO_PARITY_BITS 4
74
75#define LEGO_BITS (LEGO_CHANNEL_BITS + LEGO_MODE_BITS + LEGO_COMMAND_BITS + LEGO_PARITY_BITS)
76
77#define LEGO_HEADER_MARK 158 // 6 cycles
78#define LEGO_HEADER_SPACE 1026 // 39 cycles
79
80#define LEGO_BIT_MARK 158 // 6 cycles
81#define LEGO_ONE_SPACE 553 // 21 cycles
82#define LEGO_ZERO_SPACE 263 // 10 cycles
83
84#define LEGO_AVERAGE_DURATION 11000 // LEGO_HEADER_MARK + LEGO_HEADER_SPACE + 16 * 600 + 158
85
86#define LEGO_AUTO_REPEAT_PERIOD_MIN 110000 // Every frame is auto repeated 5 times.
87#define LEGO_AUTO_REPEAT_PERIOD_MAX 230000 // space for channel 3
88

92void IRsend::sendLegoPowerFunctions(uint16_t aRawData, bool aDoSend5Times) {
93 sendLegoPowerFunctions(aRawData, (aRawData >> (LEGO_MODE_BITS + LEGO_COMMAND_BITS + LEGO_PARITY_BITS)) & 0x3, aDoSend5Times);

96/*
97 * Here we process the structured data, and call the send raw data function
98 */
99void IRsend::sendLegoPowerFunctions(uint8_t aChannel, uint8_t aCommand, uint8_t aMode, bool aDoSend5Times) {
100 aChannel &= 0x0F; // allow toggle and escape bits too
101 aCommand &= 0x0F;
102 aMode &= 0x0F;
103 uint8_t tParity = 0xF ^ aChannel ^ aMode ^ aCommand;
104 // send 4 bit channel, 4 bit mode, 4 bit command, 4 bit parity
105 uint16_t tRawData = (((aChannel << LEGO_MODE_BITS) | aMode) << (LEGO_COMMAND_BITS + LEGO_PARITY_BITS))
106 | (aCommand << LEGO_PARITY_BITS) | tParity;
107 sendLegoPowerFunctions(tRawData, aChannel, aDoSend5Times);
108}

110void IRsend::sendLegoPowerFunctions(uint16_t aRawData, uint8_t aChannel, bool aDoSend5Times) {
111 enableIROut(38);
112
113 IR_DEBUG_PRINT("sendLego aRawData=0x");
114 IR_DEBUG_PRINTLN(aRawData, HEX);
115
116 aChannel &= 0x03; // we have 4 channels
117
118 uint_fast8_t tNumberOfCommands = 1;
119 if (aDoSend5Times) {
120 tNumberOfCommands = 5;
121 }
122// required for repeat timing, see http://www.hackvandedam.nl/blog/?page_id=559
123 uint8_t tRepeatPeriod = (110 - (LEGO_AVERAGE_DURATION / MICROS_IN_ONE_MILLI)) + (aChannel * 40); // from 100 to 220
124
125 while (tNumberOfCommands > 0) {
126
127 // Header
128 mark(LEGO_HEADER_MARK);
129 space(LEGO_HEADER_SPACE);
130
131 sendPulseDistanceWidthData(LEGO_BIT_MARK, LEGO_ONE_SPACE, LEGO_BIT_MARK, LEGO_ZERO_SPACE, aRawData, LEGO_BITS, PROTOCOL_IS_MSB_FIRST,
132 SEND_STOP_BIT);
133
134 tNumberOfCommands--;
135 // skip last delay!
136 if (tNumberOfCommands > 0) {
137 // send repeated command with a fixed space gap
138 delay(tRepeatPeriod);
139 }
140 }
141}

143/*
144 * Mode is stored in the upper nibble of command
145 */
146bool IRrecv::decodeLegoPowerFunctions() {
147
148 // Check header "mark"
149 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], LEGO_HEADER_MARK)) {
150 // no debug output, since this check is mainly to determine the received protocol
151 return false;
152 }
153
154 // Check we have enough data - +4 for initial gap, start bit mark and space + stop bit mark
155 if (decodedIRData.rawDataPtr->rawlen != (2 * LEGO_BITS) + 4) {
156 IR_DEBUG_PRINT("LEGO: ");
157 IR_DEBUG_PRINT("Data length=");
158 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
159 IR_DEBUG_PRINTLN(" is not 36");
160 return false;
161 }
162 // Check header "space"
163 if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], LEGO_HEADER_SPACE)) {
164 IR_DEBUG_PRINT("LEGO: ");
165 IR_DEBUG_PRINTLN("Header space length is wrong");
166 return false;
167 }
168
169 if (!decodePulseDistanceData(LEGO_BITS, 3, LEGO_BIT_MARK, LEGO_ONE_SPACE, LEGO_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
170 IR_DEBUG_PRINT("LEGO: ");
171 IR_DEBUG_PRINTLN("Decode failed");
172 return false;
173 }
174
175 // Stop bit
176 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * LEGO_BITS)], LEGO_BIT_MARK)) {
177 IR_DEBUG_PRINT("LEGO: ");
178 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
179 return false;
180 }
181
182 // Success
183 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
184 uint16_t tDecodedValue = decodedIRData.decodedRawData;
185 uint8_t tToggleEscapeChannel = tDecodedValue >> (LEGO_MODE_BITS + LEGO_COMMAND_BITS + LEGO_PARITY_BITS);
186 uint8_t tMode = (tDecodedValue >> (LEGO_COMMAND_BITS + LEGO_PARITY_BITS)) & 0xF;
187 uint8_t tData = (tDecodedValue >> LEGO_PARITY_BITS) & 0xF; // lego calls this field "data"
188 uint8_t tParityReceived = tDecodedValue & 0xF;
189
190 // This is parity as defined in the specifications
191 // But in some scans I saw 0x9 ^ .. as parity formula
192 uint8_t tParityComputed = 0xF ^ tToggleEscapeChannel ^ tMode ^ tData;
193
194 // parity check
195 if (tParityReceived != tParityComputed) {
196 IR_DEBUG_PRINT("LEGO: ");
197 IR_DEBUG_PRINT("Parity is not correct. expected=0x");
198 IR_DEBUG_PRINT(tParityComputed, HEX);
199 IR_DEBUG_PRINT(" received=0x");
200 IR_DEBUG_PRINT(tParityReceived, HEX);
201 IR_DEBUG_PRINT(", raw=0x");
202 IR_DEBUG_PRINT(tDecodedValue, HEX);
203 IR_DEBUG_PRINT(", 3 nibbles are 0x");
204 IR_DEBUG_PRINT(tToggleEscapeChannel, HEX);
205 IR_DEBUG_PRINT(", 0x");
206 IR_DEBUG_PRINT(tMode, HEX);
207 IR_DEBUG_PRINT(", 0x");
208 IR_DEBUG_PRINTLN(tData, HEX);
209 // might not be an error, so just continue
210 decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_MSB_FIRST;
211 }
212
213 /*
214 * Check for autorepeat (should happen 4 times for one press)
215 */
216 if (decodedIRData.rawDataPtr->rawbuf[0] < (LEGO_AUTO_REPEAT_PERIOD_MAX / MICROS_PER_TICK)) {
217 decodedIRData.flags |= IRDATA_FLAGS_IS_AUTO_REPEAT;
218 }
219 decodedIRData.address = tToggleEscapeChannel;
220 decodedIRData.command = tData | tMode << LEGO_COMMAND_BITS;
221 decodedIRData.protocol = LEGO_PF;
222 decodedIRData.numberOfBits = LEGO_BITS;
223
224 return true;
225}
