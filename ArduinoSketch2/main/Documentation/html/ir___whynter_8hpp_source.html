48bool IRrecv::decodeWhynter() {
49
50 // Check we have the right amount of data (68). The +4 is for initial gap, start bit mark and space + stop bit mark.
51 if (decodedIRData.rawDataPtr->rawlen != (2 * WHYNTER_BITS) + 4) {
52 return false;
53 }
54
55 // Sequence begins with a bit mark and a zero space
56 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], WHYNTER_BIT_MARK)
57 || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], WHYNTER_HEADER_SPACE)) {
58 IR_DEBUG_PRINT(F("Whynter: "));
59 IR_DEBUG_PRINTLN(F("Header mark or space length is wrong"));
60 return false;
61 }
62
63 if (!decodePulseDistanceData(WHYNTER_BITS, 3, WHYNTER_BIT_MARK, WHYNTER_ONE_SPACE, WHYNTER_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
64 return false;
65 }
66
67 // trailing mark / stop bit
68 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * WHYNTER_BITS)], WHYNTER_BIT_MARK)) {
69 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
70 return false;
71 }
72
73 // Success
74 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
75 decodedIRData.numberOfBits = WHYNTER_BITS;
76 decodedIRData.protocol = WHYNTER;
