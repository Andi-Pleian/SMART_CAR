140// S H U Z U

146// LSB first, 1 start bit + 16 bit address + 8 bit command + 1 stop bit.
147#define SHUZU_ADDRESS_BITS 16 // 16 bit address
148#define SHUZU_COMMAND_BITS 8 // Command
149
150#define SHUZU_BITS (SHUZU_ADDRESS_BITS + SHUZU_COMMAND_BITS) // The number of bits in the protocol
151#define SHUZU_UNIT 560
152
153#define SHUZU_HEADER_MARK (16 * SHUZU_UNIT) // The length of the Header:Mark
154#define SHUZU_HEADER_SPACE (8 * SHUZU_UNIT) // The length of the Header:Space
155
156#define SHUZU_BIT_MARK SHUZU_UNIT // The length of a Bit:Mark
157#define SHUZU_ONE_SPACE (3 * SHUZU_UNIT) // The length of a Bit:Space for 1's
158#define SHUZU_ZERO_SPACE SHUZU_UNIT // The length of a Bit:Space for 0's
159
160#define SHUZU_REPEAT_HEADER_SPACE (4 * SHUZU_UNIT) // 2250
161
162#define SHUZU_REPEAT_SPACE 45000
163
164#define SHUZU_OTHER 1234 // Other things you may need to define

168void IRsend::sendShuzu(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
169 // Set IR carrier frequency
170 enableIROut(38);
171
172 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
173 while (tNumberOfCommands > 0) {
174
175 // Header
176 mark(SHUZU_HEADER_MARK);
177 space(SHUZU_HEADER_SPACE);
178
179 // Address (device and subdevice)
180 sendPulseDistanceWidthData(SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_BIT_MARK, SHUZU_ZERO_SPACE, aAddress,
181 SHUZU_ADDRESS_BITS, PROTOCOL_IS_LSB_FIRST); // false -> LSB first
182
183 // Command + stop bit
184 sendPulseDistanceWidthData(SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_BIT_MARK, SHUZU_ZERO_SPACE, aCommand,
185 SHUZU_COMMAND_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT); // false, true -> LSB first, stop bit
186
187 tNumberOfCommands--;
188 // skip last delay!
189 if (tNumberOfCommands > 0) {
190 // send repeated command in a fixed raster
191 delay(SHUZU_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
192 }
193 }
194}

198/*
199 * First check for right data length
200 * Next check start bit
201 * Next try the decode
202 * Last check stop bit
203 */
204bool IRrecv::decodeShuzu() {
205
206 // Check we have the right amount of data (28). The +4 is for initial gap, start bit mark and space + stop bit mark
207 if (decodedIRData.rawDataPtr->rawlen != (2 * SHUZU_BITS) + 4) {
208 // no debug output, since this check is mainly to determine the received protocol
209 return false;
210 }
211
212 // Check header "space"
213 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], SHUZU_HEADER_MARK) || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], SHUZU_HEADER_SPACE)) {
214 IR_DEBUG_PRINT("Shuzu: ");
215 IR_DEBUG_PRINTLN("Header mark or space length is wrong");
216 return false;
217 }
218
219 // false -> LSB first
220 if (!decodePulseDistanceData(SHUZU_BITS, 3, SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
221 IR_DEBUG_PRINT(F("Shuzu: "));
222 IR_DEBUG_PRINTLN(F("Decode failed"));
223 return false;
224 }
225
226 // Stop bit
227 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * SHUZU_BITS)], SHUZU_BIT_MARK)) {
228 IR_DEBUG_PRINT(F("Shuzu: "));
229 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
230 return false;
231 }
232
233 // Success
234// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
235 uint8_t tCommand = decodedIRData.decodedRawData >> SHUZU_ADDRESS_BITS; // upper 8 bits of LSB first value
236 uint8_t tAddress = decodedIRData.decodedRawData & 0xFFFF; // lowest 16 bit of LSB first value
237
238 /*
239 * Check for repeat
240 */
241 if (decodedIRData.rawDataPtr->rawbuf[0] < ((SHUZU_REPEAT_SPACE + (SHUZU_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
242 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
243 }
244 decodedIRData.command = tCommand;
245 decodedIRData.address = tAddress;
246 decodedIRData.numberOfBits = SHUZU_BITS;
247 decodedIRData.protocol = LG; // we have no SHUZU code
248
249 return true;
250}
