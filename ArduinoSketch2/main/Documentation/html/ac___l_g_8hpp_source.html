46/*
47 * LG remote measurements: Type AKB73315611, Ver1.1 from 2011.03.01
48 * Internal crystal: 4 MHz
49 * Header: 8.9 ms mark 4.15 ms space
50 * Data: 500 / 540 and 500 / 1580;
51 * Clock is nor synchronized with gate so you have 19 and sometimes 19 and a spike pulses for mark
52 * Duty: 9 us on 17 us off => around 33 % duty
53 * NO REPEAT: If value like temperature has changed during long press, the last value is send at button release
54 * If you do a double press -tested with the fan button-, the next value can be sent after 118 ms
55 */
56#define SIZE_OF_FAN_SPEED_MAPPING_TABLE 4
57const int AC_FAN_TOWER[SIZE_OF_FAN_SPEED_MAPPING_TABLE] = { 0, 4, 6, 6 }; // last dummy entry to avoid out of bounds access
58const int AC_FAN_WALL[SIZE_OF_FAN_SPEED_MAPPING_TABLE] = { 0, 2, 4, 5 }; // 0 -> low, 4 high, 5 -> cycle

91/*
92 * Send repeat
93 * Repeat commands should be sent in a 110 ms raster.
94 * @param aCommand one of LG_COMMAND_OFF, LG_COMMAND_ON etc.
95 */
96bool Aircondition_LG::sendCommandAndParameter(char aCommand, int aParameter) {
97 // Commands without parameter
98 switch (aCommand) {
99 case LG_COMMAND_OFF: // off
100 sendIRCommand(LG_POWER_DOWN);
101 PowerIsOn = false;
102 return true;
103
104 case LG_COMMAND_ON: // on
105 PowerIsOn = false; // set to false in order to suppress on bit
106 sendTemperatureFanSpeedAndMode();
107 return true;
108
109 case LG_COMMAND_JET:
110 IR_DEBUG_PRINTLN(F("Send jet on"));
111 sendIRCommand(LG_JET_ON);
112 return true;
113
114 case LG_COMMAND_LIGHT:
115 sendIRCommand(LG_LIGHT);
116 return true;
117
118 case LG_COMMAND_CLEAR_ALL:
119 sendIRCommand(LG_CLEAR_ALL);
120 return true;
121
122 case LG_COMMAND_TEMPERATURE_PLUS:
123 if (18 <= Temperature && Temperature <= 29) {
124 Temperature++;
125 sendTemperatureFanSpeedAndMode();
126 } else {
127 return false;
128 }
129 return true;
130
131 case LG_COMMAND_TEMPERATURE_MINUS:
132 if (19 <= Temperature && Temperature <= 30) {
133 Temperature--;
134 sendTemperatureFanSpeedAndMode();
135 } else {
136 return false;
137 }
138 return true;
139
140 }
141
142 PowerIsOn = true;
143
144 /*
145 * Now the commands which require a parameter
146 */
147 if (aParameter < 0) {
148 IR_DEBUG_PRINT(F("Error: Parameter is less than 0"));
149 return false;
150 }
151 switch (aCommand) {
152
153 case LG_COMMAND_MODE:
154 Mode = aParameter + '0';
155 sendTemperatureFanSpeedAndMode();
156 break;
157
158 case LG_COMMAND_SWING:
159 IR_DEBUG_PRINT(F("Send air swing="));
160 IR_DEBUG_PRINTLN(aParameter);
161 if (ACIsWallType) {
162 if (aParameter) {
163 sendIRCommand(LG_WALL_SWING_ON);
164 } else {
165 sendIRCommand(LG_WALL_SWING_OFF);
166 }
167 } else {
168 if (aParameter) {
169 sendIRCommand(LG_SWING_ON);
170 } else {
171 sendIRCommand(LG_SWING_OFF);
172 }
173 }
174 break;
175
176 case LG_COMMAND_AUTO_CLEAN:
177 IR_DEBUG_PRINT(F("Send auto clean="));
178 IR_DEBUG_PRINTLN(aParameter);
179 if (aParameter) {
180 sendIRCommand(LG_AUTO_CLEAN_ON);
181 } else {
182 sendIRCommand(LG_AUTO_CLEAN_OFF);
183 }
184 break;
185
186 case LG_COMMAND_ENERGY:
187 IR_DEBUG_PRINT(F("Send energy saving on="));
188 IR_DEBUG_PRINTLN(aParameter);
189 if (aParameter) {
190 sendIRCommand(LG_ENERGY_SAVING_ON);
191 } else {
192 sendIRCommand(LG_ENERGY_SAVING_OFF);
193 }
194 break;
195
196 case LG_COMMAND_FAN_SPEED:
197 if (aParameter < SIZE_OF_FAN_SPEED_MAPPING_TABLE) {
198 FanIntensity = aParameter;
199 sendTemperatureFanSpeedAndMode();
200 } else {
201 return false;
202 }
203 break;
204
205 case LG_COMMAND_TEMPERATURE:
206 if (18 <= aParameter && aParameter <= 30) {
207 Temperature = aParameter;
208 sendTemperatureFanSpeedAndMode();
209 } else {
210 return false;
211 }
212 break;
213
214 case LG_COMMAND_SLEEP:
215 // 420 = maximum I have recorded
216 if (aParameter <= 420) {
217 sendIRCommand(LG_SLEEP + aParameter);
218 } else {
219 return false;
220 }
221 break;
222
223 case LG_COMMAND_TIMER_ON:
224 // 1440 = minutes of a day
225 if (aParameter <= 1439) {
226 sendIRCommand(LG_TIMER_ON + aParameter);
227 } else {
228 return false;
229 }
230 break;
231
232 case LG_COMMAND_TIMER_OFF:
233 if (aParameter <= 1439) {
234 sendIRCommand(LG_TIMER_OFF + aParameter);
235 } else {
236 return false;
237 }
238 break;
239
240 default:
241 return false;
242 }
243 return true;
244}

259void Aircondition_LG::sendTemperatureFanSpeedAndMode() {
260
261 uint8_t tTemperature = Temperature;
262 IR_INFO_PRINT(F("Send temperature="));
263 IR_INFO_PRINT(tTemperature);
264 IR_INFO_PRINT(F(" fan intensity="));
265 IR_INFO_PRINT(FanIntensity);
266 IR_INFO_PRINT(F(" mode="));
267 IR_INFO_PRINTLN((char )Mode);
268
269 WordUnion tIRCommand;
270 tIRCommand.UWord = 0;
271
272 // Temperature is coded in the upper nibble of the LowByte
273 tIRCommand.UByte.LowByte = ((tTemperature - 15) << 4); // 16 -> 0x00, 18 -> 0x30, 30 -> 0xF0
274
275 // Fan intensity is coded in the lower nibble of the LowByte
276 if (ACIsWallType) {
277 tIRCommand.UByte.LowByte |= AC_FAN_WALL[FanIntensity];
278 } else {
279 tIRCommand.UByte.LowByte |= AC_FAN_TOWER[FanIntensity];
280 }
281
282 switch (Mode) {
283 case AC_MODE_COOLING:
284 tIRCommand.UByte.HighByte = LG_MODE_COOLING >> 8;
285 break;
286 case AC_MODE_HEATING:
287 tIRCommand.UByte.HighByte = LG_MODE_HEATING >> 8;
288 break;
289 case AC_MODE_AUTO:
290 tIRCommand.UByte.HighByte = LG_MODE_AUTO >> 8;
291 break;
292 case AC_MODE_FAN:
293 tTemperature = 18;
294 tIRCommand.UByte.HighByte = LG_MODE_FAN >> 8;
295 break;
296 case AC_MODE_DEHUMIDIFIYING:
297 tIRCommand.UWord = LG_MODE_DEHUMIDIFIYING;
298 break;
299 default:
300 break;
301 }
302 if (!PowerIsOn) {
303 // switch on requires masked bit
304 tIRCommand.UByte.HighByte &= ~(LG_SWITCH_ON_MASK >> 8);
305 }
306 PowerIsOn = true;
307
308 sendIRCommand(tIRCommand.UWord);
309}
