45// SSSS AAA MMM SSSS U U N N GGGG
46// S A A M M M S U U NN N G
47// SSS AAAAA M M M SSS U U N N N G GG
48// S A A M M S U U N NN G G
49// SSSS A A M M SSSS UUU N N GGG
50//==============================================================================
51// see http://www.hifi-remote.com/wiki/index.php?title=DecodeIR#Samsung
52// https://www.mikrocontroller.net/articles/IRMP_-_english#SAMSUNG32
53// LSB first, 1 start bit + 16 bit address + 16,32,20 bit data + 1 stop bit.
54// repeats are like NEC but with 2 stop bits
55
56#define SAMSUNG_ADDRESS_BITS 16
57#define SAMSUNG_COMMAND16_BITS 16
58#define SAMSUNG_COMMAND32_BITS 32
59#define SAMSUNG_BITS (SAMSUNG_ADDRESS_BITS + SAMSUNG_COMMAND16_BITS)
60#define SAMSUNG48_BITS (SAMSUNG_ADDRESS_BITS + SAMSUNG_COMMAND32_BITS)
61
62#define SAMSUNG_UNIT 553 // 21 periods of 38 kHz (552,631)
63#define SAMSUNG_HEADER_MARK (8 * SAMSUNG_UNIT) // 4400
64#define SAMSUNG_HEADER_SPACE (8 * SAMSUNG_UNIT) // 4400
65#define SAMSUNG_BIT_MARK SAMSUNG_UNIT
66#define SAMSUNG_ONE_SPACE (3 * SAMSUNG_UNIT) // 1650
67#define SAMSUNG_ZERO_SPACE SAMSUNG_UNIT
68
69#define SAMSUNG_AVERAGE_DURATION 55000 // SAMSUNG_HEADER_MARK + SAMSUNG_HEADER_SPACE + 32 * 2,5 * SAMSUNG_UNIT + SAMSUNG_UNIT // 2.5 because we assume more zeros than ones
70#define SAMSUNG_REPEAT_DURATION (SAMSUNG_HEADER_MARK + SAMSUNG_HEADER_SPACE + SAMSUNG_BIT_MARK + SAMSUNG_ZERO_SPACE + SAMSUNG_BIT_MARK)
71#define SAMSUNG_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.

78void IRsend::sendSamsungRepeat() {
79 enableIROut(SAMSUNG_KHZ); // 38 kHz
80 mark(SAMSUNG_HEADER_MARK);
81 space(SAMSUNG_HEADER_SPACE);
82 mark(SAMSUNG_BIT_MARK);
83 space(SAMSUNG_ZERO_SPACE);

87void IRsend::sendSamsung(uint16_t aAddress, uint16_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
88 if (aIsRepeat) {
89 sendSamsungRepeat();
90 return;
91 }
92
93 // Set IR carrier frequency
94 enableIROut(SAMSUNG_KHZ);
95
96 // Header
97 mark(SAMSUNG_HEADER_MARK);
98 space(SAMSUNG_HEADER_SPACE);
99
100 // Address
101 sendPulseDistanceWidthData(SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_BIT_MARK, SAMSUNG_ZERO_SPACE, aAddress,
102 SAMSUNG_ADDRESS_BITS, PROTOCOL_IS_LSB_FIRST);
103
104 // Command
105
106 // send 8 command bits and then 8 inverted command bits LSB first
107 aCommand = aCommand & 0xFF;
108 aCommand = ((~aCommand) << 8) | aCommand;
109
110 sendPulseDistanceWidthData(SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_BIT_MARK, SAMSUNG_ZERO_SPACE, aCommand,
111 SAMSUNG_COMMAND16_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
112
113 for (uint_fast8_t i = 0; i < aNumberOfRepeats; ++i) {
114 // send repeat in a 110 ms raster
115 if (i == 0) {
116 delay((SAMSUNG_REPEAT_PERIOD - SAMSUNG_AVERAGE_DURATION) / MICROS_IN_ONE_MILLI);
117 } else {
118 delay((SAMSUNG_REPEAT_PERIOD - SAMSUNG_REPEAT_DURATION) / MICROS_IN_ONE_MILLI);
119 }
120 // send repeat
121 sendSamsungRepeat();
122 }
123}

126bool IRrecv::decodeSamsung() {
127
128 // Check we have enough data (68). The +4 is for initial gap, start bit mark and space + stop bit mark
129 if (decodedIRData.rawDataPtr->rawlen != ((2 * SAMSUNG_BITS) + 4)
130 && decodedIRData.rawDataPtr->rawlen != ((2 * SAMSUNG48_BITS) + 4) && (decodedIRData.rawDataPtr->rawlen != 6)) {
131 IR_DEBUG_PRINT(F("Samsung: "));
132 IR_DEBUG_PRINT("Data length=");
133 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
134 IR_DEBUG_PRINTLN(" is not 68 or 100 or 6");
135 return false;
136 }
137
138 // Check header "mark" + "space"
139 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], SAMSUNG_HEADER_MARK)
140 || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], SAMSUNG_HEADER_SPACE)) {
141 IR_DEBUG_PRINT("Samsung: ");
142 IR_DEBUG_PRINTLN("Header mark or space length is wrong");
143
144 return false;
145 }
146
147 // Check for repeat
148 if (decodedIRData.rawDataPtr->rawlen == 6) {
149 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT| IRDATA_FLAGS_IS_LSB_FIRST;
150 decodedIRData.address = lastDecodedAddress;
151 decodedIRData.command = lastDecodedCommand;
152 decodedIRData.protocol = SAMSUNG;
153 return true;
154 }
155
156 if (decodedIRData.rawDataPtr->rawlen == (2 * SAMSUNG48_BITS) + 4) {
157 /*
158 * Samsung48
159 */
160 // decode address
161 if (!decodePulseDistanceData(SAMSUNG_ADDRESS_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
162 IR_DEBUG_PRINT("Samsung: ");
163 IR_DEBUG_PRINTLN("Decode failed");
164 return false;
165 }
166 decodedIRData.address = decodedIRData.decodedRawData;
167
168 // decode 32 bit command
169 if (!decodePulseDistanceData(SAMSUNG_COMMAND32_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
170 IR_DEBUG_PRINT("Samsung: ");
171 IR_DEBUG_PRINTLN("Decode failed");
172 return false;
173 }
174
175 // Success
176// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
177 LongUnion tValue;
178 tValue.ULong = decodedIRData.decodedRawData;
179 // receive 2 * (8 bits then 8 inverted bits) LSB first
180 if (tValue.UByte.HighByte != (uint8_t) (~tValue.UByte.MidHighByte)
181 && tValue.UByte.MidLowByte != (uint8_t) (~tValue.UByte.LowByte)) {
182 decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
183 }
184 decodedIRData.command = tValue.UByte.HighByte << 8 | tValue.UByte.MidLowByte;
185 decodedIRData.numberOfBits = SAMSUNG48_BITS;
186
187 } else {
188 /*
189 * Samsung32
190 */
191 if (!decodePulseDistanceData(SAMSUNG_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
192 IR_DEBUG_PRINT("Samsung: ");
193 IR_DEBUG_PRINTLN("Decode failed");
194 return false;
195 }
196 LongUnion tValue;
197 tValue.ULong = decodedIRData.decodedRawData;
198 decodedIRData.address = tValue.UWord.LowWord;
199
200 if (tValue.UByte.MidHighByte == (uint8_t) (~tValue.UByte.HighByte)) {
201 // 8 bit command protocol
202 decodedIRData.command = tValue.UByte.MidHighByte; // first 8 bit
203 } else {
204 // 16 bit command protocol
205 decodedIRData.command = tValue.UWord.HighWord; // first 16 bit
206 }
207 decodedIRData.numberOfBits = SAMSUNG_BITS;
208 }
209
210 decodedIRData.protocol = SAMSUNG;
211
212 return true;
213}

215bool IRrecv::decodeSAMSUNG(decode_results *aResults) {
216 unsigned int offset = 1; // Skip first space
217
218 // Initial mark
219 if (!matchMark(aResults->rawbuf[offset], SAMSUNG_HEADER_MARK)) {
220 return false;
221 }
222 offset++;
223
224// Check for repeat -- like a NEC repeat
225 if ((aResults->rawlen == 4) && matchSpace(aResults->rawbuf[offset], 2250)
226 && matchMark(aResults->rawbuf[offset + 1], SAMSUNG_BIT_MARK)) {
227 aResults->bits = 0;
228 aResults->value = 0xFFFFFFFF;
229 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
230 decodedIRData.protocol = SAMSUNG;
231 return true;
232 }
233 if (aResults->rawlen < (2 * SAMSUNG_BITS) + 4) {
234 return false;
235 }
236
237// Initial space
238 if (!matchSpace(aResults->rawbuf[offset], SAMSUNG_HEADER_SPACE)) {
239 return false;
240 }
241 offset++;
242
243 if (!decodePulseDistanceData(SAMSUNG_BITS, offset, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
244 return false;
245 }
246
247// Success
248 aResults->value = decodedIRData.decodedRawData;
249 aResults->bits = SAMSUNG_BITS;
250 aResults->decode_type = SAMSUNG;
251 decodedIRData.protocol = SAMSUNG;
252 return true;
253}
