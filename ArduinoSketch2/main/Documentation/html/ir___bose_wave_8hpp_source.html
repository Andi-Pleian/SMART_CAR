20//==============================================================================
21// BBBB OOO SSSS EEEEE
22// B B O O S E
23// BB B O O SSS EEEE
24// B B O O S E
25// BBBB OOO SSSS EEEEE
26//==============================================================================
27// see http://lirc.sourceforge.net/remotes/bose/WAVERADIO
28//
29// Support for Bose Wave Radio CD initially provided by https://github.com/uvotguy.
30//
31// As seen on my trusty oscilloscope, there is no repeat code. Instead, when I
32// press and hold a button on my remote, it sends a command, makes a 51.2ms space,
33// and resends the command, etc, etc.
34
35// LSB first, 1 start bit + 8 bit data + 8 bit inverted data + 1 stop bit.
36#define BOSEWAVE_BITS 16 // Command and inverted command
37
38#define BOSEWAVE_HEADER_MARK 1060
39#define BOSEWAVE_HEADER_SPACE 1450
40#define BOSEWAVE_BIT_MARK 534
41#define BOSEWAVE_ONE_SPACE 468
42#define BOSEWAVE_ZERO_SPACE 1447
43
44#define BOSEWAVE_REPEAT_SPACE 52000

48void IRsend::sendBoseWave(uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
49 // Set IR carrier frequency
50 enableIROut(BOSEWAVE_KHZ); // 38 kHz
51
52 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
53 while (tNumberOfCommands > 0) {
54
55 // Header
56 mark(BOSEWAVE_HEADER_MARK);
57 space(BOSEWAVE_HEADER_SPACE);
58 // send 8 command bits and then 8 inverted command bits LSB first
59 uint16_t tData = ((~aCommand) << 8) | aCommand;
60
61 sendPulseDistanceWidthData(BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_BIT_MARK, BOSEWAVE_ZERO_SPACE, tData,
62 BOSEWAVE_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
63
64 tNumberOfCommands--;
65 // skip last delay!
66 if (tNumberOfCommands > 0) {
67 // send repeated command with a fixed space gap
68 delay( BOSEWAVE_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
69 }
70 }
71}

74bool IRrecv::decodeBoseWave() {
75
76 // Check header "mark"
77 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], BOSEWAVE_HEADER_MARK)) {
78 // no debug output, since this check is mainly to determine the received protocol
79 return false;
80 }
81
82 // Check we have enough data +4 for initial gap, start bit mark and space + stop bit mark
83 if (decodedIRData.rawDataPtr->rawlen != (2 * BOSEWAVE_BITS) + 4) {
84 IR_DEBUG_PRINT("Bose: ");
85 IR_DEBUG_PRINT("Data length=");
86 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
87 IR_DEBUG_PRINTLN(" is not 36");
88 return false;
89 }
90 // Check header "space"
91 if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], BOSEWAVE_HEADER_SPACE)) {
92 IR_DEBUG_PRINT("Bose: ");
93 IR_DEBUG_PRINTLN("Header space length is wrong");
94 return false;
95 }
96
97 if (!decodePulseDistanceData(BOSEWAVE_BITS, 3, BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
98 IR_DEBUG_PRINT("Bose: ");
99 IR_DEBUG_PRINTLN("Decode failed");
100 return false;
101 }
102
103 // Stop bit
104 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * BOSEWAVE_BITS)], BOSEWAVE_BIT_MARK)) {
105 IR_DEBUG_PRINT("Bose: ");
106 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
107 return false;
108 }
109
110 // Success
111// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
112 uint16_t tDecodedValue = decodedIRData.decodedRawData;
113 uint8_t tCommandNotInverted = tDecodedValue & 0xFF;
114 uint8_t tCommandInverted = tDecodedValue >> 8;
115 // parity check for command. Use this variant to avoid compiler warning "comparison of promoted ~unsigned with unsigned [-Wsign-compare]"
116 if ((tCommandNotInverted ^ tCommandInverted) != 0xFF) {
117 IR_DEBUG_PRINT("Bose: ");
118 IR_DEBUG_PRINT("Command and inverted command check failed");
119 return false;
120 }
121
122 // check for repeat
123 if (decodedIRData.rawDataPtr->rawbuf[0] < ((BOSEWAVE_REPEAT_SPACE + (BOSEWAVE_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
124 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
125 }
126
127 decodedIRData.command = tCommandNotInverted;
128 decodedIRData.protocol = BOSEWAVE;
129 decodedIRData.numberOfBits = BOSEWAVE_BITS;
130
131 return true;
132}
