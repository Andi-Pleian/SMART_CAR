44//==============================================================================
45// PPPP AAA N N AAA SSSS OOO N N IIIII CCCC
46// P P A A NN N A A S O O NN N I C
47// PPPP AAAAA N N N AAAAA SSS O O N N N I C
48// P A A N NN A A S O O N NN I C
49// P A A N N A A SSSS OOO N N IIIII CCCC
50//==============================================================================
51// see: http://www.hifi-remote.com/johnsfine/DecodeIR.html#Panasonic and http://www.hifi-remote.com/johnsfine/DecodeIR.html#Kaseikyo
52// IRP notation: {37k,432}<1,-1|1,-3>(8,-4,M:8,N:8,X:4,D:4,S:8,F:8,G:8,1,-173)+ {X=M:4:0^M:4:4^N:4:0^N:4:4}
53// see: http://www.remotecentral.com/cgi-bin/mboard/rc-pronto/thread.cgi?26152
54// The first two (8-bit) bytes are always 2 and 32 (These identify Panasonic within the Kaseikyo standard)
55// The next two bytes are 4 independent 4-bit fields or Device and Subdevice
56// The second to last byte is the function and the last byte is xor of the three bytes before it.
57// 0_______ 1_______ 2______ 3_______ 4_______ 5
58// 76543210 76543210 76543210 76543210 76543210 76543210
59// 00000010 00100000 Dev____ Sub Dev Fun____ XOR( B2, B3, B4)
60// LSB first, start bit + 16 Vendor + 4 Parity(of vendor) + 4 Genre1 + 4 Genre2 + 10 Command + 2 ID + 8 Parity + stop bit
61// We reduce it to: start bit + 16 Vendor + 16 Address + 8 Command + 8 Parity + stop bit
62//
63#define KASEIKYO_VENDOR_ID_BITS 16
64#define KASEIKYO_VENDOR_ID_PARITY_BITS 4
65#define KASEIKYO_ADDRESS_BITS 12
66#define KASEIKYO_COMMAND_BITS 8
67#define KASEIKYO_PARITY_BITS 8
68#define KASEIKYO_BITS (KASEIKYO_VENDOR_ID_BITS + KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_ADDRESS_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS)
69#define KASEIKYO_UNIT 432 // 16 pulses of 37 kHz (432,432432) - Pronto 0x70 / 0x10
70
71#define KASEIKYO_HEADER_MARK (8 * KASEIKYO_UNIT) // 3456
72#define KASEIKYO_HEADER_SPACE (4 * KASEIKYO_UNIT) // 1728
73
74#define KASEIKYO_BIT_MARK KASEIKYO_UNIT
75#define KASEIKYO_ONE_SPACE (3 * KASEIKYO_UNIT) // 1296
76#define KASEIKYO_ZERO_SPACE KASEIKYO_UNIT
77
78#define KASEIKYO_AVERAGE_DURATION 56000
79#define KASEIKYO_REPEAT_PERIOD 130000
80#define KASEIKYO_REPEAT_SPACE (KASEIKYO_REPEAT_PERIOD - KASEIKYO_AVERAGE_DURATION) // 74 ms
81
82// for old decoder
83#define KASEIKYO_DATA_BITS 32

86/*
87 * Send with LSB first
88 * Address is sub-device << 8 + device
89 */
90void IRsend::sendKaseikyo(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, uint16_t aVendorCode) {
91 // Set IR carrier frequency
92 enableIROut(KASEIKYO_KHZ); // 37 kHz
93
94 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
95 while (tNumberOfCommands > 0) {
96
97 // Header
98 mark(KASEIKYO_HEADER_MARK);
99 space(KASEIKYO_HEADER_SPACE);
100
101 // Vendor ID
102 sendPulseDistanceWidthData(KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_BIT_MARK, KASEIKYO_ZERO_SPACE, aVendorCode,
103 KASEIKYO_VENDOR_ID_BITS, PROTOCOL_IS_LSB_FIRST);
104
105 // Vendor Parity
106 uint8_t tVendorParity = aVendorCode ^ (aVendorCode >> 8);
107 tVendorParity = (tVendorParity ^ (tVendorParity >> 4)) & 0xF;
108
109 LongUnion tSendValue;
110 tSendValue.UWord.LowWord = aAddress << KASEIKYO_VENDOR_ID_PARITY_BITS;
111 tSendValue.UByte.LowByte |= tVendorParity; // set low nibble to parity
112 tSendValue.UByte.MidHighByte = aCommand;
113 tSendValue.UByte.HighByte = aCommand ^ tSendValue.UByte.LowByte ^ tSendValue.UByte.MidLowByte; // Parity
114
115 // Send address (device and subdevice) + command + parity + Stop bit
116 sendPulseDistanceWidthData(KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_BIT_MARK, KASEIKYO_ZERO_SPACE, tSendValue.ULong,
117 KASEIKYO_ADDRESS_BITS + KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS,
118 PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
119
120 tNumberOfCommands--;
121 // skip last delay!
122 if (tNumberOfCommands > 0) {
123 // send repeated command in a fixed raster
124 delay(KASEIKYO_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
125 }
126 }
127}

167bool IRrecv::decodeKaseikyo() {
168
169 decode_type_t tProtocol;
170 // Check we have enough data (100)- +4 for initial gap, start bit mark and space + stop bit mark
171 if (decodedIRData.rawDataPtr->rawlen != ((2 * KASEIKYO_BITS) + 4)) {
172 IR_DEBUG_PRINT(F("Kaseikyo: "));
173 IR_DEBUG_PRINT("Data length=");
174 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
175 IR_DEBUG_PRINTLN(" is not 100");
176 return false;
177 }
178
179 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], KASEIKYO_HEADER_MARK)) {
180 IR_DEBUG_PRINT("Kaseikyo: ");
181 IR_DEBUG_PRINTLN("Header mark length is wrong");
182 return false;
183 }
184
185 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[2], KASEIKYO_HEADER_SPACE)) {
186 IR_DEBUG_PRINT("Kaseikyo: ");
187 IR_DEBUG_PRINTLN("Header space length is wrong");
188 return false;
189 }
190
191 // decode Vendor ID
192 if (!decodePulseDistanceData(KASEIKYO_VENDOR_ID_BITS, 3, KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_ZERO_SPACE,
193 PROTOCOL_IS_LSB_FIRST)) {
194 IR_DEBUG_PRINT("Kaseikyo: ");
195 IR_DEBUG_PRINTLN("Vendor ID decode failed");
196 return false;
197 }
198
199 uint16_t tVendorId = decodedIRData.decodedRawData;
200 if (tVendorId == PANASONIC_VENDOR_ID_CODE) {
201 tProtocol = PANASONIC;
202 } else if (tVendorId == SHARP_VENDOR_ID_CODE) {
203 tProtocol = KASEIKYO_SHARP;
204 } else if (tVendorId == DENON_VENDOR_ID_CODE) {
205 tProtocol = KASEIKYO_DENON;
206 } else if (tVendorId == JVC_VENDOR_ID_CODE) {
207 tProtocol = KASEIKYO_JVC;
208 } else if (tVendorId == MITSUBISHI_VENDOR_ID_CODE) {
209 tProtocol = KASEIKYO_MITSUBISHI;
210 } else {
211 tProtocol = KASEIKYO;
212 }
213
214 // Vendor Parity
215 uint8_t tVendorParity = tVendorId ^ (tVendorId >> 8);
216 tVendorParity = (tVendorParity ^ (tVendorParity >> 4)) & 0xF;
217
218 // decode address (device and subdevice) + command + parity
219 if (!decodePulseDistanceData(
220 KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_ADDRESS_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS,
221 3 + (2 * KASEIKYO_VENDOR_ID_BITS), KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE,
222 KASEIKYO_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
223 IR_DEBUG_PRINT("Kaseikyo: ");
224 IR_DEBUG_PRINTLN("Address, command + parity decode failed");
225 return false;
226 }
227
228 // Success
229// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
230 LongUnion tValue;
231 tValue.ULong = decodedIRData.decodedRawData;
232 decodedIRData.address = (tValue.UWord.LowWord >> KASEIKYO_VENDOR_ID_PARITY_BITS); // remove vendor parity
233 decodedIRData.command = tValue.UByte.MidHighByte;
234 uint8_t tParity = tValue.UByte.LowByte ^ tValue.UByte.MidLowByte ^ tValue.UByte.MidHighByte;
235
236 if (tVendorParity != (tValue.UByte.LowByte & 0xF)) {
237 IR_DEBUG_PRINT("Kaseikyo: ");
238 IR_DEBUG_PRINT("4 bit VendorID Parity is not correct. expected=0x");
239 IR_DEBUG_PRINT(tVendorParity, HEX);
240 IR_DEBUG_PRINT(" received=0x");
241 IR_DEBUG_PRINT(decodedIRData.decodedRawData, HEX);
242 IR_DEBUG_PRINT(" VendorID=0x");
243 IR_DEBUG_PRINTLN(tVendorId, HEX);
244 decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
245 }
246
247 if (tProtocol != KASEIKYO) {
248 decodedIRData.flags |= IRDATA_FLAGS_EXTRA_INFO;
249 // Include vendor ID in address
250 decodedIRData.extra |= tVendorId;
251 }
252
253 if (tValue.UByte.HighByte != tParity) {
254 IR_DEBUG_PRINT("Kaseikyo: ");
255 IR_DEBUG_PRINT("8 bit Parity is not correct. expected=0x");
256 IR_DEBUG_PRINT(tParity, HEX);
257 IR_DEBUG_PRINT(" received=0x");
258 IR_DEBUG_PRINT(decodedIRData.decodedRawData >> KASEIKYO_COMMAND_BITS, HEX);
259 IR_DEBUG_PRINT(" address=0x");
260 IR_DEBUG_PRINT(decodedIRData.address, HEX);
261 IR_DEBUG_PRINT(" command=0x");
262 IR_DEBUG_PRINTLN(decodedIRData.command, HEX);
263 decodedIRData.flags |= IRDATA_FLAGS_PARITY_FAILED;
264 }
265
266 // check for repeat
267 if (decodedIRData.rawDataPtr->rawbuf[0] < (KASEIKYO_REPEAT_PERIOD / MICROS_PER_TICK)) {
268 decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
269 }
270
271 decodedIRData.protocol = tProtocol;
272
273 decodedIRData.numberOfBits = KASEIKYO_BITS;
274
275 return true;
276}

281bool IRrecv::decodePanasonicMSB(decode_results *aResults) {
282 unsigned int offset = 1;
283
284 if (aResults->rawlen < (2 * KASEIKYO_BITS) + 2) {
285 return false;
286 }
287
288 if (!matchMark(aResults->rawbuf[offset], KASEIKYO_HEADER_MARK)) {
289 return false;
290 }
291 offset++;
292 if (!matchMark(aResults->rawbuf[offset], KASEIKYO_HEADER_SPACE)) {
293 return false;
294 }
295 offset++;
296
297 // decode address
298 if (!decodePulseDistanceData(KASEIKYO_ADDRESS_BITS + KASEIKYO_DATA_BITS, offset, KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE,
299 KASEIKYO_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
300 return false;
301 }
302
303 aResults->bits = KASEIKYO_BITS;
304 aResults->value = decodedIRData.decodedRawData;
305 aResults->address = PANASONIC_VENDOR_ID_CODE;
306 aResults->decode_type = PANASONIC;
307 decodedIRData.protocol = PANASONIC;
308
309 return true;
310}
