44// DDDD EEEEE N N OOO N N
45// D D E NN N O O NN N
46// D D EEE N N N O O N N N
47// D D E N NN O O N NN
48// DDDD EEEEE N N OOO N N
49//==============================================================================
50// SSSS H H AAA RRRR PPPP
51// S H H A A R R P P
52// SSS HHHHH AAAAA RRRR PPPP
53// S H H A A R R P
54// SSSS H H A A R R P
55//==============================================================================
56// Denon publish all their IR codes:
57// https://www.google.co.uk/search?q=DENON+MASTER+IR+Hex+Command+Sheet
58// -> http://assets.denon.com/documentmaster/us/denon%20master%20ir%20hex.xls
59
60// Having looked at the official Denon Pronto sheet and reverse engineered
61// the timing values from it, it is obvious that Denon have a range of
62// different timings and protocols ...the values here work for my AVR-3801 Amp!
63
64// MSB first, no start bit, 5 address + 8 command + 2 frame + 1 stop bit - each frame 2 times
65//
66#define DENON_ADDRESS_BITS 5
67#define DENON_COMMAND_BITS 8
68#define DENON_FRAME_BITS 2 // 00/10 for 1. frame Denon/Sharp, inverted for autorepeat frame
69
70#define DENON_BITS (DENON_ADDRESS_BITS + DENON_COMMAND_BITS + DENON_FRAME_BITS) // 15 - The number of bits in the command
71#define DENON_UNIT 260
72
73#define DENON_BIT_MARK DENON_UNIT // The length of a Bit:Mark
74#define DENON_ONE_SPACE (7 * DENON_UNIT) // 1820 // The length of a Bit:Space for 1's
75#define DENON_ZERO_SPACE (3 * DENON_UNIT) // 780 // The length of a Bit:Space for 0's
76
77#define DENON_AUTO_REPEAT_SPACE 45000 // Every frame is auto repeated with a space period of 45 ms and the command inverted.
78#define DENON_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
79
80// for old decoder
81#define DENON_HEADER_MARK DENON_UNIT // The length of the Header:Mark
82#define DENON_HEADER_SPACE (3 * DENON_UNIT) // 780 // The length of the Header:Space
83

97void IRsend::sendDenon(uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aSendSharp) {
98 // Set IR carrier frequency
99 enableIROut(DENON_KHZ); // 38 kHz
100
101 // Shift command and add frame marker
102 uint16_t tCommand = aCommand << DENON_FRAME_BITS; // the lowest bits are 00 for Denon and 10 for Sharp
103 if (aSendSharp) {
104 tCommand |= 0x02;
105 }
106 uint16_t tData = tCommand | ((uint16_t) aAddress << (DENON_COMMAND_BITS + DENON_FRAME_BITS));
107 uint16_t tInvertedData = ((~tCommand) & 0x3FF) | (uint16_t) aAddress << (DENON_COMMAND_BITS + DENON_FRAME_BITS);
108
109 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
110 while (tNumberOfCommands > 0) {
111
112 // Data
113 sendPulseDistanceWidthData(DENON_BIT_MARK, DENON_ONE_SPACE, DENON_BIT_MARK, DENON_ZERO_SPACE, tData, DENON_BITS, PROTOCOL_IS_MSB_FIRST,
114 SEND_STOP_BIT);
115
116 // Inverted autorepeat frame
117 delay(DENON_AUTO_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
118 sendPulseDistanceWidthData(DENON_BIT_MARK, DENON_ONE_SPACE, DENON_BIT_MARK, DENON_ZERO_SPACE, tInvertedData, DENON_BITS,
119 PROTOCOL_IS_MSB_FIRST, SEND_STOP_BIT);
120
121 tNumberOfCommands--;
122 // skip last delay!
123 if (tNumberOfCommands > 0) {
124 // send repeated command with a fixed space gap
125 delay( DENON_AUTO_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
126 }
127 }
128}

136bool IRrecv::decodeDenon() {
137
138 // we have no start bit, so check for the exact amount of data bits
139 // Check we have the right amount of data (32). The + 2 is for initial gap + stop bit mark
140 if (decodedIRData.rawDataPtr->rawlen != (2 * DENON_BITS) + 2) {
141 IR_DEBUG_PRINT(F("Denon: "));
142 IR_DEBUG_PRINT("Data length=");
143 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
144 IR_DEBUG_PRINTLN(" is not 32");
145 return false;
146 }
147
148 // Read the bits in
149 if (!decodePulseDistanceData(DENON_BITS, 1, DENON_BIT_MARK, DENON_ONE_SPACE, DENON_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
150 IR_DEBUG_PRINT("Denon: ");
151 IR_DEBUG_PRINTLN("Decode failed");
152 return false;
153 }
154
155 // Check for stop mark
156 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[(2 * DENON_BITS) + 1], DENON_HEADER_MARK)) {
157 IR_DEBUG_PRINT("Denon: ");
158 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
159 return false;
160 }
161
162 // Success
163 decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
164 uint8_t tFrameBits = decodedIRData.decodedRawData & 0x03;
165 decodedIRData.command = decodedIRData.decodedRawData >> DENON_FRAME_BITS;
166 decodedIRData.address = decodedIRData.command >> DENON_COMMAND_BITS;
167 uint8_t tCommand = decodedIRData.command & 0xFF;
168 decodedIRData.command = tCommand;
169
170 // check for autorepeated inverted command
171 if (decodedIRData.rawDataPtr->rawbuf[0] < ((DENON_AUTO_REPEAT_SPACE + (DENON_AUTO_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
172 repeatCount++;
173 if (tFrameBits == 0x3 || tFrameBits == 0x1) {
174 // We are in the auto repeated frame with the inverted command
175 decodedIRData.flags = IRDATA_FLAGS_IS_AUTO_REPEAT | IRDATA_FLAGS_IS_MSB_FIRST;
176 // Check parity of consecutive received commands. There is no parity in one data set.
177 uint8_t tLastCommand = lastDecodedCommand;
178 if (tLastCommand != (uint8_t) (~tCommand)) {
179 decodedIRData.flags |= IRDATA_FLAGS_PARITY_FAILED;
180 }
181 // always take non inverted command
182 decodedIRData.command = tLastCommand;
183 }
184 if (repeatCount > 1) {
185 decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
186 }
187 } else {
188 repeatCount = 0;
189 }
190
191 decodedIRData.numberOfBits = DENON_BITS;
192 if (tFrameBits == 1 || tFrameBits == 2) {
193 decodedIRData.protocol = SHARP;
194 } else {
195 decodedIRData.protocol = DENON;
196 }
197 return true;
198}

200bool IRrecv::decodeDenonOld(decode_results *aResults) {
201
202 // Check we have the right amount of data
203 if (decodedIRData.rawDataPtr->rawlen != 1 + 2 + (2 * DENON_BITS) + 1) {
204 return false;
205 }
206
207 // Check initial Mark+Space match
208 if (!matchMark(aResults->rawbuf[1], DENON_HEADER_MARK)) {
209 return false;
210 }
211
212 if (!matchSpace(aResults->rawbuf[2], DENON_HEADER_SPACE)) {
213 return false;
214 }
215
216 // Read the bits in
217 if (!decodePulseDistanceData(DENON_BITS, 3, DENON_BIT_MARK, DENON_ONE_SPACE, DENON_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
218 return false;
219 }
220
221 // Success
222 aResults->value = decodedIRData.decodedRawData;
223 aResults->bits = DENON_BITS;
224 aResults->decode_type = DENON;
225 decodedIRData.protocol = DENON;
226 return true;
227}

230 // Set IR carrier frequency
231 enableIROut(DENON_KHZ);
232#if !(defined(__AVR_ATtiny25__) || defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__) || defined(__AVR_ATtiny87__) || defined(__AVR_ATtiny167__))

239 space(DENON_HEADER_SPACE);
240

243 SEND_STOP_BIT);

245}
