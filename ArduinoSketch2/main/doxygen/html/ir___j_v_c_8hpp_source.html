43//==============================================================================
44// JJJJJ V V CCCC
45// J V V C
46// J V V C
47// J J V V C
48// J V CCCC
49//==============================================================================
50// https://www.sbprojects.net/knowledge/ir/jvc.php
51// http://www.hifi-remote.com/johnsfine/DecodeIR.html#JVC
52// IRP: {38k,525}<1,-1|1,-3>(16,-8,(D:8,F:8,1,-45)+)
53// LSB first, 1 start bit + 8 bit address + 8 bit command + 1 stop bit.
54// The JVC protocol repeats by skipping the header mark and space -> this leads to a poor repeat detection for JVC protocol.
55#define JVC_ADDRESS_BITS 8 // 8 bit address
56#define JVC_COMMAND_BITS 8 // Command
57
58#define JVC_BITS (JVC_ADDRESS_BITS + JVC_COMMAND_BITS) // 16 - The number of bits in the protocol
59#define JVC_UNIT 526 // 20 periods of 38 kHz (526.315789)
60
61#define JVC_HEADER_MARK (16 * JVC_UNIT) // 8400
62#define JVC_HEADER_SPACE (8 * JVC_UNIT) // 4200
63
64#define JVC_BIT_MARK JVC_UNIT // The length of a Bit:Mark
65#define JVC_ONE_SPACE (3 * JVC_UNIT) // 1578 - The length of a Bit:Space for 1's
66#define JVC_ZERO_SPACE JVC_UNIT // The length of a Bit:Space for 0's
67
68#define JVC_REPEAT_SPACE (uint16_t)(45 * JVC_UNIT) // 23625 - Commands are repeated with a distance of 23 ms for as long as the key on the remote control is held down.
69
70//+=============================================================================
71// JVC does NOT repeat by sending a separate code (like NEC does).
72// The JVC protocol repeats by skipping the header.
73//
74
75void IRsend::sendJVC(uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
76 // Set IR carrier frequency
77 enableIROut(JVC_KHZ); // 38 kHz
78
79 // Header
80 mark(JVC_HEADER_MARK);
81 space(JVC_HEADER_SPACE);
82
83 uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
84 while (tNumberOfCommands > 0) {
85
86 // Address + command
87 sendPulseDistanceWidthData(JVC_BIT_MARK, JVC_ONE_SPACE, JVC_BIT_MARK, JVC_ZERO_SPACE,
88 aAddress | (aCommand << JVC_ADDRESS_BITS), JVC_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
89
90 tNumberOfCommands--;
91 // skip last delay!
92 if (tNumberOfCommands > 0) {
93 // send repeated command in a fixed raster
94 delay(JVC_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
95 }
96 }
97}

99/*
100 * First check for right data length
101 * Next check start bit
102 * Next try the decode
103 */
104bool IRrecv::decodeJVC() {
105
106 // Check we have the right amount of data (36 or 34). The +4 is for initial gap, start bit mark and space + stop bit mark. +2 is for repeats
107 if (decodedIRData.rawDataPtr->rawlen != ((2 * JVC_BITS) + 4) && decodedIRData.rawDataPtr->rawlen != ((2 * JVC_BITS) + 2)) {
108 IR_DEBUG_PRINT(F("JVC: "));
109 IR_DEBUG_PRINT("Data length=");
110 IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
111 IR_DEBUG_PRINTLN(" is not 34 or 36");
112 return false;
113 }
114
115 if (decodedIRData.rawDataPtr->rawlen == ((2 * JVC_BITS) + 2)) {
116 /*
117 * Check for repeat
118 * Check leading space and first and last mark length
119 */
120 if (decodedIRData.rawDataPtr->rawbuf[0] < ((JVC_REPEAT_SPACE + (JVC_REPEAT_SPACE / 4) / MICROS_PER_TICK))
121 && matchMark(decodedIRData.rawDataPtr->rawbuf[1], JVC_BIT_MARK)
122 && matchMark(decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen - 1], JVC_BIT_MARK)) {
123 /*
124 * We have a repeat here, so do not check for start bit
125 */
126 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
127 decodedIRData.address = lastDecodedAddress;
128 decodedIRData.command = lastDecodedCommand;
129 decodedIRData.protocol = JVC;
130 }
131 } else {
132
133 // Check header "mark" and "space"
134 if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], JVC_HEADER_MARK)
135 || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], JVC_HEADER_SPACE)) {
136 IR_DEBUG_PRINT("JVC: ");
137 IR_DEBUG_PRINTLN("Header mark or space length is wrong");
138 return false;
139 }
140
141 if (!decodePulseDistanceData(JVC_BITS, 3, JVC_BIT_MARK, JVC_ONE_SPACE, JVC_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
142 IR_DEBUG_PRINT(F("JVC: "));
143 IR_DEBUG_PRINTLN(F("Decode failed"));
144 return false;
145 }
146
147 // Success
148// decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
149 uint8_t tCommand = decodedIRData.decodedRawData >> JVC_ADDRESS_BITS; // upper 8 bits of LSB first value
150 uint8_t tAddress = decodedIRData.decodedRawData & 0xFF; // lowest 8 bit of LSB first value
151
152 decodedIRData.command = tCommand;
153 decodedIRData.address = tAddress;
154 decodedIRData.numberOfBits = JVC_BITS;
155 decodedIRData.protocol = JVC;
156 }
157
158 return true;
159}

161bool IRrecv::decodeJVCMSB(decode_results *aResults) {
162 unsigned int offset = 1; // Skip first space
163
164 // Check for repeat
165 if ((aResults->rawlen - 1 == 33) && matchMark(aResults->rawbuf[offset], JVC_BIT_MARK)
166 && matchMark(aResults->rawbuf[aResults->rawlen - 1], JVC_BIT_MARK)) {
167 aResults->bits = 0;
168 aResults->value = 0xFFFFFFFF;
169 decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
170 decodedIRData.protocol = JVC;
171 return true;
172 }
173
174 // Initial mark
175 if (!matchMark(aResults->rawbuf[offset], JVC_HEADER_MARK)) {
176 return false;
177 }
178 offset++;
179
180 // Check we have enough data - +3 for start bit mark and space + stop bit mark
181 if (aResults->rawlen <= (2 * JVC_BITS) + 3) {
182 IR_DEBUG_PRINT("Data length=");
183 IR_DEBUG_PRINT(aResults->rawlen);
184 IR_DEBUG_PRINTLN(" is too small. >= 36 is required.");
185
186 return false;
187 }
188
189 // Initial space
190 if (!matchSpace(aResults->rawbuf[offset], JVC_HEADER_SPACE)) {
191 return false;
192 }
193 offset++;
194
195 if (!decodePulseDistanceData(JVC_BITS, offset, JVC_BIT_MARK, JVC_ONE_SPACE, JVC_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
196 return false;
197 }
198
199 // Stop bit
200 if (!matchMark(aResults->rawbuf[offset + (2 * JVC_BITS)], JVC_BIT_MARK)) {
201 IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
202 return false;
203 }
204
205 // Success
206 aResults->value = decodedIRData.decodedRawData;
207 aResults->bits = JVC_BITS;
208 aResults->decode_type = JVC;
209 decodedIRData.protocol = JVC;
210
211 return true;
212}
